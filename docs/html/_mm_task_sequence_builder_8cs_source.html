159 public class MmTaskSequenceHandler<T> where T : class, IMmTaskInfo, new()
160 {
161 private readonly MmTaskManager<T> _taskManager;
162 private T _previousTask;
163
164 public Action<T> OnTaskStart { get; set; }
165 public Action<T> OnTaskComplete { get; set; }
166 public Action<T, T> OnTaskChange { get; set; }
167 public Action OnSequenceStart { get; set; }
168 public Action OnSequenceComplete { get; set; }
169 public Func<T, bool> TaskFilter { get; set; }
170 public bool AutoAdvance { get; set; }
171
172 public MmTaskSequenceHandler(MmTaskManager<T> taskManager)
173 {
174 _taskManager = taskManager;
175 }
176
180 public void Start()
181 {
182 OnSequenceStart?.Invoke();
183 _taskManager.ProceedToFirstTask();
184 NotifyTaskStart();
185 }
186
190 public void Next()
191 {
192 NotifyTaskComplete();
193
194 _previousTask = _taskManager.CurrentTaskInfo;
195 _taskManager.ProceedToNextTask();
196
197 if (_taskManager.CurrentTaskInfo == null)
198 {
199 OnSequenceComplete?.Invoke();
200 }
201 else
202 {
203 OnTaskChange?.Invoke(_previousTask, _taskManager.CurrentTaskInfo);
204 NotifyTaskStart();
205 }
206 }
207
211 public void CompleteCurrentTask()
212 {
213 NotifyTaskComplete();
214
215 if (AutoAdvance)
216 {
217 Next();
218 }
219 }
220
221 private void NotifyTaskStart()
222 {
223 var current = _taskManager.CurrentTaskInfo;
224 if (current != null && (TaskFilter == null || TaskFilter(current)))
225 {
226 OnTaskStart?.Invoke(current);
227 }
228 }
229
230 private void NotifyTaskComplete()
231 {
232 var current = _taskManager.CurrentTaskInfo;
233 if (current != null && (TaskFilter == null || TaskFilter(current)))
234 {
235 OnTaskComplete?.Invoke(current);
236 }
237 }
238
242 public bool HasNextTask => _taskManager.NextTaskInfo != null;
243
247 public T CurrentTask => _taskManager.CurrentTaskInfo;
248 }
