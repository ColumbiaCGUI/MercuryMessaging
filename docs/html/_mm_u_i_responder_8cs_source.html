65 public override void MmInvoke(MmMessage message)
66 {
67 int methodValue = (int)message.MmMethod;
68
69 // UI messages are in range 100-199
70 if (methodValue >= 100 && methodValue < 200)
71 {
72 DispatchUIMessage(message);
73 return;
74 }
75
76 // All other messages go to base class
77 base.MmInvoke(message);
78 }

83 private void DispatchUIMessage(MmMessage message)
