67 public override void MmInvoke(MmMessage msg)
68 {
69 var type = msg.MmMethod;
70
71 switch (type)
72 {
73 case MmMethod.NoOp:
74 break;
75 case MmMethod.SetActive:
76 var messageBool = (MmMessageBool) msg;
77 SetActive(messageBool.value);
78 break;
79 case MmMethod.Refresh:
80 // Handle both MmMessageTransformList and plain MmMessage
81 // (DSL's Refresh() creates plain MmMessage, direct calls may use MmMessageTransformList)
82 if (msg is MmMessageTransformList messageTransform)
83 Refresh(messageTransform.transforms);
84 else
85 Refresh(new List<MmTransform>()); // Empty list for parameter-less Refresh
86 break;
87 case MmMethod.Initialize:
88 Initialize();
89 break;
90 case MmMethod.Switch:
91 var messageString = (MmMessageString) msg;
92 Switch(messageString.value);
93 break;
94 case MmMethod.Complete:
95 var messageCompleteBool = (MmMessageBool) msg;
96 Complete(messageCompleteBool.value);
97 break;
98 case MmMethod.TaskInfo:
99 var messageSerializable = (MmMessageSerializable) msg;
100 if (messageSerializable.value is IMmSerializable taskValue)
101 ApplyTaskInfo(taskValue);
102 break;
103 case MmMethod.Message:
104 ReceivedMessage(msg);
105 break;
106 case MmMethod.MessageBool:
107 ReceivedMessage((MmMessageBool) msg);
108 break;
109 case MmMethod.MessageByteArray:
110 ReceivedMessage((MmMessageByteArray) msg);
111 break;
112 case MmMethod.MessageFloat:
113 ReceivedMessage((MmMessageFloat) msg);
114 break;
115 case MmMethod.MessageInt:
116 ReceivedMessage((MmMessageInt) msg);
117 break;
118 case MmMethod.MessageSerializable:
119 ReceivedMessage((MmMessageSerializable) msg);
120 break;
121 case MmMethod.MessageString:
122 ReceivedMessage((MmMessageString) msg);
123 break;
124 case MmMethod.MessageTransform:
125 ReceivedMessage((MmMessageTransform) msg);
126 break;
127 case MmMethod.MessageTransformList:
128 ReceivedMessage((MmMessageTransformList) msg);
129 break;
130 case MmMethod.MessageVector3:
131 ReceivedMessage((MmMessageVector3) msg);
132 break;
133 case MmMethod.MessageVector4:
134 ReceivedMessage((MmMessageVector4) msg);
135 break;
136 case MmMethod.MessageGameObject:
137 ReceivedMessage((MmMessageGameObject) msg);
138 break;
139 case MmMethod.MessageQuaternion:
140 ReceivedMessage((MmMessageQuaternion)msg);
141 break;
142 default:
143 Debug.Log(msg.MmMethod.ToString());
144 throw new ArgumentOutOfRangeException();
145 }
146 }
