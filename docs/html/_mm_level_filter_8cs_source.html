50 [Flags]
51 public enum MmLevelFilter
52 {
53 // Original filters (bits 0-2)
54 Self = 1 << 0, // Bit 0: Self node only
55 Child = 1 << 1, // Bit 1: Direct children
56 Parent = 1 << 2, // Bit 2: Direct parents
57
58 // Phase 2.1: Advanced Routing (bits 3-7)
59 Siblings = 1 << 3, // Bit 3: Same-parent nodes (lateral routing)
60 Cousins = 1 << 4, // Bit 4: Parent's-sibling's-children
61 Descendants = 1 << 5, // Bit 5: Recursive all children (Child + grandchildren + ...)
62 Ancestors = 1 << 6, // Bit 6: Recursive all parents (Parent + grandparents + ...)
63 Custom = 1 << 7, // Bit 7: User-defined predicate (requires MmRoutingOptions.CustomFilter)
64 }
