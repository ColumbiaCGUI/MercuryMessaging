137 public void EnterNext()
138 {
139 if (!isTransitioning) return;

143 if (LogMessage != null)
144 LogMessage(String.Format(">>>,{0},{1},{2}", Name, Current, Next));
145 Previous = Current;
146 Current = Next;
147 Next = default(T);
148
149 //Console.WriteLine(String.Format("Entering {0}",Current));
150 if (this[Current].Enter != null) this[Current].Enter();
