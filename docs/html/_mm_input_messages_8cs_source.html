133 public class MmInput6DOFMessage : MmMessage
134 {
136 public MmHandedness Hand;
137
139 public Vector3 Position;
140
142 public Quaternion Rotation;
143
145 public Vector3 Velocity;
146
148 public Vector3 AngularVelocity;
149
151 public bool IsTracked;
152
153 public MmInput6DOFMessage() : base((MmMethod)MmInputMethod.Input6DOF, (MmMessageType)MmInputMessageType.Input6DOF)
154 {
155 Rotation = Quaternion.identity;
156 IsTracked = true;
157 }
158
159 public MmInput6DOFMessage(MmHandedness hand, Vector3 position, Quaternion rotation,
160 Vector3 velocity = default, Vector3 angularVelocity = default, bool isTracked = true,
161 MmMetadataBlock metadataBlock = null)
162 : base((MmMethod)MmInputMethod.Input6DOF, (MmMessageType)MmInputMessageType.Input6DOF, metadataBlock)
163 {
164 Hand = hand;
165 Position = position;
166 Rotation = rotation;
167 Velocity = velocity;
168 AngularVelocity = angularVelocity;
169 IsTracked = isTracked;
170 }
171
172 public MmInput6DOFMessage(MmInput6DOFMessage message) : base(message)
173 {
174 Hand = message.Hand;
175 Position = message.Position;
176 Rotation = message.Rotation;
177 Velocity = message.Velocity;
178 AngularVelocity = message.AngularVelocity;
179 IsTracked = message.IsTracked;
180 }
181
182 public override MmMessage Copy() => new MmInput6DOFMessage(this);
183
184 public override int Deserialize(object[] data)
185 {
186 int index = base.Deserialize(data);
187 Hand = (MmHandedness)(int)data[index++];
188 Position = new Vector3((float)data[index++], (float)data[index++], (float)data[index++]);
189 Rotation = new Quaternion((float)data[index++], (float)data[index++], (float)data[index++], (float)data[index++]);
190 Velocity = new Vector3((float)data[index++], (float)data[index++], (float)data[index++]);
191 AngularVelocity = new Vector3((float)data[index++], (float)data[index++], (float)data[index++]);
192 IsTracked = (bool)data[index++];
193 return index;
194 }
195
196 public override object[] Serialize()
197 {
198 return base.Serialize().Concat(new object[] {
199 (int)Hand,
200 Position.x, Position.y, Position.z,
201 Rotation.x, Rotation.y, Rotation.z, Rotation.w,
202 Velocity.x, Velocity.y, Velocity.z,
203 AngularVelocity.x, AngularVelocity.y, AngularVelocity.z,
204 IsTracked
205 }).ToArray();
206 }
