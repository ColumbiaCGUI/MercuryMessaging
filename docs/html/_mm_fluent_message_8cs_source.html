628 [MethodImpl(MethodImplOptions.AggressiveInlining)]
629 public void Execute()
630 {
631 if (_relay == null)
632 {
633 Debug.LogError("MmFluentMessage: Cannot execute without a relay node");
634 return;
635 }
636
637 // Optimization 2.2: Fast path for simple messages (no predicates, no target collection)
638 // This is the most common case - avoids branching for predicates/collection checks
639 if (_predicates == null && !_needsTargetCollection)
640 {
641 // Optimization 2.3: Use pre-allocated metadata for common cases
642 MmMetadataBlock metadata = _hasCustomFilters
643 ? CreateMetadata()
644 : DefaultMetadata;
645 ExecuteStandard(metadata);
646 return;
647 }
648
649 // Build metadata block for non-fast-path cases
650 MmMetadataBlock fullMetadata = CreateMetadata();
651
652 // If predicates exist, use predicate-based routing
653 if (_predicates != null && _predicates.Count > 0)
654 {
655 ExecuteWithPredicates(fullMetadata);
656 return;
657 }
658
659 // Optimization 2.1: Use cached flag instead of computing 4 bitwise ANDs + 3 ORs
660 // The flag is set in ToXxx() methods when the level filter changes
661 if (_needsTargetCollection)
662 {
663 ExecuteWithTargetCollection();
664 return;
665 }
666
667 // Standard routing path (SelfAndChildren, Child, Self)
668 ExecuteStandard(fullMetadata);
669 }

701 private void ExecuteWithTargetCollection()
702 {
703 // Collect all potential targets based on level filter
704 var targets = CollectTargets();
705
706 // Send to each target with Self metadata
707 // Using Self (not SelfAndChildren) prevents double-delivery:
708 // - We explicitly send to each collected target
709 // - Self reaches responders registered with Level=Self on the target
710 // - SelfAndChildren would cause messages to propagate down, hitting children twice
711 foreach (var targetRelay in targets)
712 {
713 if (targetRelay == null || targetRelay.gameObject == null)
714 continue;
715
716 // Create Self metadata - responders are registered with Level=Self so this works
717 MmMetadataBlock targetMetadata;
718 if (_tag != MmTagHelper.Everything)
719 {
720 targetMetadata = new MmMetadataBlock(_tag, MmLevelFilter.Self, _activeFilter, _selectedFilter, _networkFilter);
721 }
722 else
723 {
724 targetMetadata = new MmMetadataBlock(MmLevelFilter.Self, _activeFilter, _selectedFilter, _networkFilter);
725 }
726
727 // Send message to this specific target
728 SendToTarget(targetRelay, targetMetadata);
729 }
730 }

735 private void ExecuteStandard(MmMetadataBlock metadata)
736 {
737 // Handle MmMessage payloads directly (from Send(MmMessage) or Broadcast(MmMessage))
738 if (_payload is MmMessage message)
739 {
740 message.MetadataBlock = metadata;
741 _relay.MmInvoke(message);
742 return;
743 }
744
745 // Handle primitive payloads
746 switch (_method)
747 {
748 case MmMethod.Initialize:
749 case MmMethod.Refresh:
750 case MmMethod.NoOp:
751 _relay.MmInvoke(_method, metadata);
752 break;
753
754 // Complete requires a bool parameter (MmBaseResponder casts to MmMessageBool)
755 case MmMethod.Complete:
756 _relay.MmInvoke(_method, _payload is bool completeBool ? completeBool : true, metadata);
757 break;
758
759 case MmMethod.SetActive:
760 case MmMethod.MessageBool:
761 if (_payload is bool boolValue)
762 _relay.MmInvoke(_method, boolValue, metadata);
763 else
764 Debug.LogError($"MmFluentMessage: Expected bool payload for {_method}");
765 break;
766
767 case MmMethod.Switch:
768 if (_payload is string switchName)
769 _relay.MmInvoke(_method, switchName, metadata);
770 else
771 Debug.LogError($"MmFluentMessage: Expected string payload for {_method}");
772 break;
773
774 case MmMethod.MessageInt:
775 if (_payload is int intValue)
776 _relay.MmInvoke(_method, intValue, metadata);
777 else
778 Debug.LogError($"MmFluentMessage: Expected int payload for {_method}");
779 break;
780
781 case MmMethod.MessageFloat:
782 if (_payload is float floatValue)
783 _relay.MmInvoke(_method, floatValue, metadata);
784 else
785 Debug.LogError($"MmFluentMessage: Expected float payload for {_method}");
786 break;
787
788 case MmMethod.MessageString:
789 if (_payload is string stringValue)
790 _relay.MmInvoke(_method, stringValue, metadata);
791 else
792 Debug.LogError($"MmFluentMessage: Expected string payload for {_method}");
793 break;
794
795 case MmMethod.MessageVector3:
796 if (_payload is Vector3 vector3Value)
797 _relay.MmInvoke(_method, vector3Value, metadata);
798 else
799 Debug.LogError($"MmFluentMessage: Expected Vector3 payload for {_method}");
800 break;
801
802 case MmMethod.MessageVector4:
803 if (_payload is Vector4 vector4Value)
804 _relay.MmInvoke(_method, vector4Value, metadata);
805 else
806 Debug.LogError($"MmFluentMessage: Expected Vector4 payload for {_method}");
807 break;
808
809 case MmMethod.MessageQuaternion:
810 if (_payload is Quaternion quaternionValue)
811 _relay.MmInvoke(_method, quaternionValue, metadata);
812 else
813 Debug.LogError($"MmFluentMessage: Expected Quaternion payload for {_method}");
814 break;
815
816 case MmMethod.MessageTransform:
817 if (_payload is Transform transformValue)
818 _relay.MmInvoke(_method, transformValue, metadata);
819 else
820 Debug.LogError($"MmFluentMessage: Expected Transform payload for {_method}");
821 break;
822
823 case MmMethod.MessageGameObject:
824 if (_payload is GameObject gameObjectValue)
825 _relay.MmInvoke(_method, gameObjectValue, metadata);
826 else
827 Debug.LogError($"MmFluentMessage: Expected GameObject payload for {_method}");
828 break;
829
830 default:
831 Debug.LogError($"MmFluentMessage: Unsupported method {_method} or invalid payload");
832 break;
833 }
834 }

841 private void ExecuteWithPredicates(MmMetadataBlock metadata)
842 {
843 // Collect all potential targets based on level filter
844 var targets = CollectTargets();
845
846 // Apply predicates and send to matching targets
847 foreach (var targetRelay in targets)
848 {
849 if (targetRelay == null || targetRelay.gameObject == null)
850 continue;
851
852 // Check if target passes all predicates
853 if (!_predicates.EvaluateAll(_relay, targetRelay.gameObject))
854 continue;
855
856 // Create Self metadata - responders are registered with Level=Self so this works
857 // Using Self (not SelfAndChildren) prevents double-delivery
858 MmMetadataBlock targetMetadata;
859 if (_tag != MmTagHelper.Everything)
860 {
861 targetMetadata = new MmMetadataBlock(_tag, MmLevelFilter.Self, _activeFilter, _selectedFilter, _networkFilter);
862 }
863 else
864 {
865 targetMetadata = new MmMetadataBlock(MmLevelFilter.Self, _activeFilter, _selectedFilter, _networkFilter);
866 }
867
868 // Send message to this specific target
869 SendToTarget(targetRelay, targetMetadata);
870 }
871
872 // Return predicate list to pool
873 MmPredicateListPool.Return(_predicates);
874 _predicates = null;
875 }

880 private List<MmRelayNode> CollectTargets()
881 {
882 var targets = new List<MmRelayNode>();
883
884 // Check if Self is included
885 bool includeSelf = (_levelFilter & MmLevelFilter.Self) != 0;
886 if (includeSelf)
887 {
888 targets.Add(_relay);
889 }
890
891 // Check for children/descendants
892 bool includeChildren = (_levelFilter & MmLevelFilter.Child) != 0;
893 bool includeDescendants = (_levelFilter & MmLevelFilter.Descendants) != 0;
894
895 if (includeChildren || includeDescendants)
896 {
897 CollectChildRelays(_relay, targets, includeDescendants);
898 }
899
900 // Check for parents/ancestors
901 bool includeParents = (_levelFilter & MmLevelFilter.Parent) != 0;
902 bool includeAncestors = (_levelFilter & MmLevelFilter.Ancestors) != 0;
903
904 if (includeParents || includeAncestors)
905 {
906 CollectParentRelays(_relay, targets, includeAncestors);
907 }
908
909 // Check for siblings
910 bool includeSiblings = (_levelFilter & MmLevelFilter.Siblings) != 0;
911 if (includeSiblings)
912 {
913 CollectSiblingRelays(_relay, targets);
914 }
915
916 return targets;
917 }

922 private void CollectChildRelays(MmRelayNode parent, List<MmRelayNode> targets, bool recursive)
923 {
924 if (parent == null || parent.RoutingTable == null)
925 return;
926
927 foreach (var item in parent.RoutingTable)
928 {
929 if (item.Level != MmLevelFilter.Child)
930 continue;
931
932 var childRelay = item.Responder?.GetRelayNode();
933 if (childRelay != null && !targets.Contains(childRelay))
934 {
935 targets.Add(childRelay);
936
937 if (recursive)
938 {
939 CollectChildRelays(childRelay, targets, true);
940 }
941 }
942 }
943 }

949 private void CollectParentRelays(MmRelayNode child, List<MmRelayNode> targets, bool recursive)

956 if (item.Level != MmLevelFilter.Parent)
957 continue;

960 if (parentRelay != null && !targets.Contains(parentRelay))
961 {
962 targets.Add(parentRelay);
963
964 if (recursive)
965 {
966 CollectParentRelays(parentRelay, targets, true);

976 private void CollectSiblingRelays(MmRelayNode node, List<MmRelayNode> targets)
977 {
978 if (node == null || node.RoutingTable == null)
979 return;
980
981 // Find parents first
982 foreach (var item in node.RoutingTable)
983 {
984 if (item.Level != MmLevelFilter.Parent)
985 continue;
986
987 var parent = item.Responder?.GetRelayNode();
988 if (parent == null || parent.RoutingTable == null)
989 continue;
990
991 // Collect parent's children (our siblings)
992 foreach (var childItem in parent.RoutingTable)
993 {
994 if (childItem.Level != MmLevelFilter.Child)
995 continue;
996
997 var siblingRelay = childItem.Responder?.GetRelayNode();
998 if (siblingRelay != null && siblingRelay != node && !targets.Contains(siblingRelay))
999 {
1000 targets.Add(siblingRelay);
1001 }
1002 }
1003 }

1010 {

1012 if (_payload is MmMessage message)
1013 {
1014 var msgCopy = message.Copy();
1015 msgCopy.MetadataBlock = metadata;
1016 target.MmInvoke(msgCopy);
1017 return;
1018 }
1019
1020 // Handle primitive payloads
1021 switch (_method)
1022 {
1023 case MmMethod.Initialize:
1024 case MmMethod.Refresh:
1025 case MmMethod.NoOp:
1026 target.MmInvoke(_method, metadata);
1027 break;
1028
1029 // Complete requires a bool parameter (MmBaseResponder casts to MmMessageBool)
1030 case MmMethod.Complete:
1031 target.MmInvoke(_method, _payload is bool completeBool ? completeBool : true, metadata);
1032 break;
1033
1034 case MmMethod.SetActive:
1035 case MmMethod.MessageBool:
1036 if (_payload is bool boolValue)
1037 target.MmInvoke(_method, boolValue, metadata);
1038 break;
1039
1040 case MmMethod.Switch:
1041 if (_payload is string switchName)
1042 target.MmInvoke(_method, switchName, metadata);
1043 break;
1044
1045 case MmMethod.MessageInt:
1046 if (_payload is int intValue)
1047 target.MmInvoke(_method, intValue, metadata);
1048 break;
1049
1050 case MmMethod.MessageFloat:
1051 if (_payload is float floatValue)
1052 target.MmInvoke(_method, floatValue, metadata);
1053 break;
1054
1055 case MmMethod.MessageString:
1056 if (_payload is string stringValue)
1057 target.MmInvoke(_method, stringValue, metadata);
1058 break;
1059
1060 case MmMethod.MessageVector3:
1061 if (_payload is Vector3 vector3Value)
1062 target.MmInvoke(_method, vector3Value, metadata);
1063 break;
1064
1065 case MmMethod.MessageVector4:
1066 if (_payload is Vector4 vector4Value)
1067 target.MmInvoke(_method, vector4Value, metadata);
