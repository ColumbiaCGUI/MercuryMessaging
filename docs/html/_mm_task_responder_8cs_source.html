110 public virtual void InitializeCompletionFSM()
111 {
112 TaskStateFSM = new FiniteStateMachine<TaskState>
113 (gameObject.name + "_TaskStateFSM")
114 {
115 LogMessage = MmLogger.LogApplication
116 };
117
118 TaskStateFSM[TaskState.Complete].Enter = delegate
119 {
120 if (_mmRelayNode != null)
121 {
122 _mmRelayNode.MmInvoke(MmMethod.Complete, true,
123 new MmMetadataBlock(MmLevelFilter.Parent));
124 }
125 };
126
127 TaskStateFSM.JumpTo(TaskState.NotComplete);
128
129 TaskStateFSM[TaskState.Complete].Exit = delegate
130 {
131 if (_mmRelayNode != null)
132 {
133 _mmRelayNode.MmInvoke(MmMethod.Complete, false,
134 new MmMetadataBlock(MmLevelFilter.Parent));
135 }
136 };
137 }
138
143 public override void Update()
144 {
145 //Debug.Log("MmTaskResponder Update " + gameObject.name);
146
147 if (TaskCompleteCheck())
148 {
149 if(TaskStateFSM.Current == TaskState.NotComplete)
150 TaskStateFSM.JumpTo(TaskState.Complete);
151 }
152 else
153 {
154 if(TaskStateFSM.Current == TaskState.Complete)
155 TaskStateFSM.JumpTo(TaskState.NotComplete);
156 }
157 }
