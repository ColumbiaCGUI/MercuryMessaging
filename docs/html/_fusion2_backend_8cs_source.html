453 private int GetOrAssignClientId(PlayerRef player)
454 {
455 if (_playerToClientId.TryGetValue(player, out int existingId))
456 return existingId;
457
458 int newId = _nextClientId++;
459 _playerToClientId[player] = newId;
460 _clientIdToPlayer[newId] = player;
461 return newId;
462 }
