63 public MmObjectPool(int maxSize = 100, Action<T> resetAction = null)
64 {

67 }

72 public void Prewarm(int count)
73 {
74 lock (_lock)
75 {
76 for (int i = 0; i < count && _pool.Count < _maxSize; i++)
77 {
78 _pool.Push(new T());
79 TotalCreated++;
80 }
81 }
82 }
83
87 public T Get()
88 {
89 TotalGets++;
90
91 lock (_lock)
92 {
93 if (_pool.Count > 0)
94 {
95 var obj = _pool.Pop();
96 // Mark poolable objects as from pool
97 if (obj is IMmPoolable poolable)
98 {
99 poolable.Reset(); // This sets IsFromPool = true for MmPooledMessage
100 }
101 return obj;
102 }
103 }
104
105 TotalCreated++;
106 var newObj = new T();
107 // Mark new poolable objects as from pool (they will be returned to pool later)
108 if (newObj is IMmPoolable newPoolable)
109 {
110 newPoolable.Reset(); // This sets IsFromPool = true for MmPooledMessage
111 }
112 return newObj;
113 }
114
118 public void Return(T obj)
119 {
120 if (obj == null) return;
121
122 TotalReturns++;
123 _resetAction?.Invoke(obj);
124
125 lock (_lock)
126 {
127 if (_pool.Count < _maxSize)
128 {
129 _pool.Push(obj);
130 }
131 // Excess objects are discarded (GC will collect them)
132 }
133 }
134
138 public void Clear()
139 {
140 lock (_lock)
141 {
142 _pool.Clear();
143 }
144 }
145
149 public string GetStats()
150 {
151 return $"Pool<{typeof(T).Name}>: Count={Count}/{_maxSize}, Created={TotalCreated}, Gets={TotalGets}, Returns={TotalReturns}, HitRate={HitRate:P1}";
152 }

159 public class MmByteArrayPool
160 {
161 // Bucket sizes: 64, 256, 1024, 4096, 16384, 65536
162 private static readonly int[] BucketSizes = { 64, 256, 1024, 4096, 16384, 65536 };
163 private readonly Stack<byte[]>[] _buckets;
164 private readonly object[] _locks;
165 private readonly int _maxArraysPerBucket;
166
170 public int TotalCreated { get; private set; }
171
175 public int TotalGets { get; private set; }
176
180 public int TotalReturns { get; private set; }
181
185 public MmByteArrayPool(int maxArraysPerBucket = 50)
186 {
187 _maxArraysPerBucket = maxArraysPerBucket;
188 _buckets = new Stack<byte[]>[BucketSizes.Length];
189 _locks = new object[BucketSizes.Length];
190
191 for (int i = 0; i < BucketSizes.Length; i++)
192 {
193 _buckets[i] = new Stack<byte[]>();
194 _locks[i] = new object();
195 }
196 }
197
201 public byte[] Get(int minSize)
202 {
203 TotalGets++;
204
205 int bucketIndex = GetBucketIndex(minSize);
206 if (bucketIndex < 0)
207 {
208 // Too large for pooling, allocate directly
209 TotalCreated++;
210 return new byte[minSize];
211 }
212
213 lock (_locks[bucketIndex])
214 {
215 if (_buckets[bucketIndex].Count > 0)
216 {
217 return _buckets[bucketIndex].Pop();
218 }
219 }
220
221 TotalCreated++;
222 return new byte[BucketSizes[bucketIndex]];
223 }
224
228 public void Return(byte[] array)
229 {
230 if (array == null) return;
231
232 TotalReturns++;
233
234 int bucketIndex = GetExactBucketIndex(array.Length);
235 if (bucketIndex < 0)
236 {
237 // Not a pooled size, let GC collect it
238 return;
239 }
240
241 // Clear sensitive data
242 Array.Clear(array, 0, array.Length);
243
244 lock (_locks[bucketIndex])
245 {
246 if (_buckets[bucketIndex].Count < _maxArraysPerBucket)
247 {
248 _buckets[bucketIndex].Push(array);
249 }
250 }
251 }
252
256 public ScopedArray GetScoped(int minSize)
257 {
258 return new ScopedArray(this, Get(minSize));
259 }
260
264 public void Clear()
265 {
266 for (int i = 0; i < _buckets.Length; i++)
267 {
268 lock (_locks[i])
269 {
270 _buckets[i].Clear();
271 }
272 }
273 }
274
275 private int GetBucketIndex(int size)
276 {
277 for (int i = 0; i < BucketSizes.Length; i++)
278 {
279 if (BucketSizes[i] >= size)
280 {
281 return i;
282 }
283 }
284 return -1; // Too large
285 }
286
287 private int GetExactBucketIndex(int size)
288 {
289 for (int i = 0; i < BucketSizes.Length; i++)
290 {
291 if (BucketSizes[i] == size)
292 {
293 return i;
294 }
295 }
296 return -1;
297 }
298
302 public string GetStats()
303 {
304 var stats = new System.Text.StringBuilder();
305 stats.AppendLine($"ByteArrayPool: Created={TotalCreated}, Gets={TotalGets}, Returns={TotalReturns}");
306 for (int i = 0; i < BucketSizes.Length; i++)
307 {
308 lock (_locks[i])
309 {
310 stats.AppendLine($" Bucket[{BucketSizes[i]}]: {_buckets[i].Count}/{_maxArraysPerBucket}");
311 }
312 }
313 return stats.ToString();
