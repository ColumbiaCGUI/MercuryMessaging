212 #region Reset Helper
213
217 private static void ResetMessage(MmMessage msg)
218 {
219 msg.MmMethod = default;
220 msg.MetadataBlock = new MmMetadataBlock();
221 msg.NetId = 0;
222 msg.root = true;
223 msg.TimeStamp = null;
224 msg.HopCount = 0;
225 msg.Handled = false; // Reset Handled flag for E1-E3 early termination feature
226 msg._isPooled = true; // Mark as pooled for return tracking
227
228 // Return VisitedNodes HashSet to pool if it exists
229 if (msg.VisitedNodes != null)
230 {
231 MmHashSetPool.Return(msg.VisitedNodes);
232 msg.VisitedNodes = null;
233 }
234 }

423 #region Return Methods
424
431 public static bool Return(MmMessage message)
432 {
433 if (message == null)
434 return false;
435
436 // Don't pool network-deserialized messages - they have different lifecycles
437 if (message.IsDeserialized)
438 return false;
439
440 switch (message.MmMessageType)
441 {
442 case MmMessageType.MmVoid:
443 _messagePool.Release(message);
444 break;
445 case MmMessageType.MmBool:
446 _boolPool.Release((MmMessageBool)message);
447 break;
448 case MmMessageType.MmInt:
449 _intPool.Release((MmMessageInt)message);
450 break;
451 case MmMessageType.MmFloat:
452 _floatPool.Release((MmMessageFloat)message);
453 break;
454 case MmMessageType.MmString:
455 _stringPool.Release((MmMessageString)message);
456 break;
457 case MmMessageType.MmVector3:
458 _vector3Pool.Release((MmMessageVector3)message);
459 break;
460 case MmMessageType.MmVector4:
461 _vector4Pool.Release((MmMessageVector4)message);
462 break;
463 case MmMessageType.MmQuaternion:
464 _quaternionPool.Release((MmMessageQuaternion)message);
465 break;
466 case MmMessageType.MmTransform:
467 _transformPool.Release((MmMessageTransform)message);
468 break;
469 case MmMessageType.MmTransformList:
470 _transformListPool.Release((MmMessageTransformList)message);
471 break;
472 case MmMessageType.MmByteArray:
473 _byteArrayPool.Release((MmMessageByteArray)message);
474 break;
475 case MmMessageType.MmSerializable:
476 _serializablePool.Release((MmMessageSerializable)message);
477 break;
478 case MmMessageType.MmGameObject:
479 _gameObjectPool.Release((MmMessageGameObject)message);
480 break;
481 default:
482 // Unknown type, don't pool
483 return false;
484 }
485
486 return true;
487 }
