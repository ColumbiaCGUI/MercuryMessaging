145 public void Add(int nodeId)
146 {
147 float now = Time.realtimeSinceStartup;
148 _addCount++;
149
150 // If node already visited, refresh its timestamp (move to end of queue)
151 if (_visitedNodes.Contains(nodeId))
152 {
153 if (_nodeMap.TryGetValue(nodeId, out var existingNode))
154 {
155 // Remove from current position
156 _evictionQueue.Remove(existingNode);
157
158 // Add to end with updated timestamp
159 var newEntry = new CacheEntry(nodeId, now);
160 var newNode = _evictionQueue.AddLast(newEntry);
161 _nodeMap[nodeId] = newNode;
162 }
163 return;
164 }
165
166 // Add new node
167 _visitedNodes.Add(nodeId);
168 var entry = new CacheEntry(nodeId, now);
169 var listNode = _evictionQueue.AddLast(entry);
170 _nodeMap[nodeId] = listNode;
171
172 // Evict old entries periodically (amortized cost)
173 if (now - _lastEvictionCheck >= EVICTION_CHECK_INTERVAL)
174 {
175 EvictOldEntries(now);
176 _lastEvictionCheck = now;
177 }
178 }

235 private void EvictOldEntries(float currentTime)
236 {
237 float evictionThreshold = currentTime - _windowSizeSeconds;
238
239 // Evict from front of queue (oldest entries)
240 while (_evictionQueue.Count > 0)
241 {
242 var oldestEntry = _evictionQueue.First.Value;
243
244 // Stop when we hit entry strictly within window
245 // Use > (not >=) so that entries at exactly the threshold are evicted
246 // This ensures 0ms window evicts all entries immediately
247 if (oldestEntry.Timestamp > evictionThreshold)
248 break;
249
250 // Remove from all data structures
251 _visitedNodes.Remove(oldestEntry.NodeId);
252 _nodeMap.Remove(oldestEntry.NodeId);
253 _evictionQueue.RemoveFirst();
254 _evictionCount++;
255 }
256 }
257
262 private int GetOldestEntryAge()
263 {
264 if (_evictionQueue.Count == 0)
265 return 0;
266
267 float now = Time.realtimeSinceStartup;
268 float oldestTime = _evictionQueue.First.Value.Timestamp;
269 return (int)((now - oldestTime) * 1000);
270 }
