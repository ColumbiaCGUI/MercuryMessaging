39 public class InterpolatorOrbit : Interpolator<Matrix4x4>
40 {
41 private float beginLength, endLength;
42
43 public override void Begin()
44 {
45 beginLength = valueBegin.GetPosition().magnitude;
46 endLength = valueEnd.GetPosition().magnitude;
47
48 base.Begin();
49 }

51 public void SetSpeed(float speed)
52 {
53 var dist = CalculateDistance();
54
55 Duration = (dist / speed);
56 }
57
58 public override void Apply()
59 {
60 //Notifier.AddMessage(StepCountTotal.ToString());
61 var f = Mathf.SmoothStep(0, 1, DurationCurrent / Duration);
62 //var f = (float)_step / _steps;
63 var l = Mathf.Lerp(beginLength, endLength, f);
64
65 var d = Vector3.Lerp(valueBegin.GetPosition(), valueEnd.GetPosition(), f);
66 d.Normalize();
67 d *= l;
68
69 ValueCurrent =
70 MatrixUtilities.CreateRotationMatrix(Quaternion.Slerp(ValueOriginal.GetRotation(), ValueTarget.GetRotation(), f))
71 * MatrixUtilities.CreateTranslationMatrix(d);
72
73 transform.position = d;
74 transform.rotation = Quaternion.Slerp(ValueOriginal.GetRotation(), ValueTarget.GetRotation(), f);
75 }
76
77 public override void Revert()
78 {
79 transform.position = ValueOriginal.GetPosition();
80 transform.rotation = ValueOriginal.GetRotation();
81 }
82
83 private float CalculateDistance()
84 {
85 // Orbit distance
86 var v1 = Utilities.Project2D(ValueOriginal.inverse.GetPosition(), Vector3.forward);
87 v1.Normalize();
88
89 var v2 = Utilities.Project2D(ValueTarget.inverse.GetPosition(), Vector3.forward);
90 v2.Normalize();
91
92 var angle = Mathf.Abs(Vector3.Angle(v1, v2));
93
94 // Rotate distance
95 v1 = Utilities.Project2D(ValueOriginal.GetPosition(), Vector3.forward);
96 v1.Normalize();
97
98 v2 = Utilities.Project2D(ValueTarget.GetPosition(), Vector3.forward);
99 v2.Normalize();
100
101 var angle2 = Mathf.Abs(Vector3.Angle(v1, v2));
102
103 if (angle2 > angle) angle = angle2;
104
105 var radius = (ValueOriginal.inverse.GetPosition().magnitude + ValueTarget.inverse.GetPosition().magnitude) / 2;
106
107 var arcDist = radius * angle;
108 var lineDist = (ValueTarget.inverse.GetPosition() - ValueOriginal.inverse.GetPosition()).magnitude;
109
110 return (lineDist > arcDist) ? lineDist : arcDist;
111 }
112 }
