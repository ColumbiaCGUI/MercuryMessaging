76 public static List<MmStateDelta> DeserializeBatch(byte[] data)
77 {
78 using (var ms = new MemoryStream(data))
79 using (var reader = new BinaryReader(ms))
80 {
81 // Read and verify header
82 ushort header = reader.ReadUInt16();
83 if (header != DeltaPacketHeader)
84 {
85 throw new InvalidDataException($"Invalid delta packet header: 0x{header:X4}");
86 }
87
88 // Read count
89 ushort count = reader.ReadUInt16();
90
91 // Read deltas
92 var deltas = new List<MmStateDelta>(count);
93 for (int i = 0; i < count; i++)
94 {
95 deltas.Add(ReadDelta(reader));
96 }
97
98 return deltas;
99 }

105 public static bool IsDeltaBatchPacket(byte[] data)
106 {
107 if (data == null || data.Length < 2) return false;
108 ushort header = (ushort)(data[0] | (data[1] << 8));
109 return header == DeltaPacketHeader;

112 private static void WriteDelta(BinaryWriter writer, MmStateDelta delta)
113 {
114 // Write delta header
115 writer.Write(delta.DeltaId);
116 writer.Write(delta.BaselineSequence);
117 writer.Write(delta.NetId);
118 writer.Write(delta.Timestamp);
119 writer.Write(delta.IsFullSnapshot);
120
121 // Write field count
122 writer.Write((ushort)delta.ChangedFields.Count);
123
124 // Write each field
125 foreach (var field in delta.ChangedFields)
126 {
127 WriteField(writer, field);
128 }
129 }

131 private static MmStateDelta ReadDelta(BinaryReader reader)
132 {
133 var delta = new MmStateDelta
134 {
135 DeltaId = reader.ReadUInt32(),
136 BaselineSequence = reader.ReadUInt32(),
137 NetId = reader.ReadUInt32(),
138 Timestamp = reader.ReadDouble(),
139 IsFullSnapshot = reader.ReadBoolean()
140 };
141
142 // Read field count
143 ushort fieldCount = reader.ReadUInt16();
144
145 // Read each field
146 delta.ChangedFields = new List<MmDeltaField>(fieldCount);
147 for (int i = 0; i < fieldCount; i++)
148 {
149 delta.ChangedFields.Add(ReadField(reader));
150 }
151
152 return delta;
153 }
154
155 private static void WriteField(BinaryWriter writer, MmDeltaField field)
156 {
157 writer.Write(field.FieldId);
158 writer.Write((byte)field.FieldType);
159 writer.Write((ushort)field.Value.Length);
160 writer.Write(field.Value);
161 }
162
163 private static MmDeltaField ReadField(BinaryReader reader)
164 {
165 ushort fieldId = reader.ReadUInt16();
166 MmDeltaFieldType fieldType = (MmDeltaFieldType)reader.ReadByte();
167 ushort valueLength = reader.ReadUInt16();
168 byte[] value = reader.ReadBytes(valueLength);
169
170 return new MmDeltaField(fieldId, fieldType, value);
171 }

294 public void Enqueue(byte[] data, MmMessagePriority priority, int targetClient, MmReliability reliability, double currentTime)
295 {
296 var msg = new MmQueuedMessage
297 {
298 Data = data,
299 Priority = priority,
300 QueuedTime = currentTime,
301 TargetClient = targetClient,
302 Reliability = reliability
303 };
304
305 lock (_lock)
306 {
307 _heap.Add(msg);
308 HeapifyUp(_heap.Count - 1);
309 }

315 public bool TryDequeue(out MmQueuedMessage message)
316 {
317 lock (_lock)
318 {
319 if (_heap.Count == 0)
320 {
321 message = default;
322 return false;
323 }
324
325 message = _heap[0];
326
327 // Move last element to top and heapify down
328 int lastIdx = _heap.Count - 1;
329 _heap[0] = _heap[lastIdx];
330 _heap.RemoveAt(lastIdx);
331
332 if (_heap.Count > 0)
333 {
334 HeapifyDown(0);
335 }
336
337 return true;
338 }
339 }
340
344 public bool TryPeek(out MmQueuedMessage message)
345 {
346 lock (_lock)
347 {
348 if (_heap.Count == 0)
349 {
350 message = default;
351 return false;
352 }
353
354 message = _heap[0];
355 return true;
356 }

362 public List<MmQueuedMessage> DequeueBatch(int maxBytes)
363 {
364 var result = new List<MmQueuedMessage>();
365 int totalBytes = 0;
366
367 lock (_lock)
368 {
369 while (_heap.Count > 0)
370 {
371 var peek = _heap[0];
372 if (totalBytes + peek.Data.Length > maxBytes && result.Count > 0)
373 {
374 break; // Would exceed limit, stop here
375 }
376
377 if (TryDequeue(out var msg))
378 {
379 result.Add(msg);
380 totalBytes += msg.Data.Length;
381 }
382 }
383 }
384
385 return result;
386 }

391 public int DropLowPriority(int maxToKeep)
392 {
393 int dropped = 0;
394
395 lock (_lock)
396 {
397 if (_heap.Count <= maxToKeep) return 0;
398
399 // Rebuild heap keeping only higher priority messages
400 var toKeep = new List<MmQueuedMessage>();
401 foreach (var msg in _heap)
402 {
403 if (msg.Priority >= MmMessagePriority.Normal || toKeep.Count < maxToKeep)
404 {
405 toKeep.Add(msg);
406 }
407 else
408 {
409 dropped++;
410 }
411 }
412
413 // Rebuild heap
414 _heap.Clear();
415 foreach (var msg in toKeep)
416 {
417 _heap.Add(msg);
418 HeapifyUp(_heap.Count - 1);
419 }
420 }
421
422 return dropped;
423 }

436 private void HeapifyUp(int index)
437 {
438 while (index > 0)
439 {
440 int parentIdx = (index - 1) / 2;
441 if (_heap[index].CompareTo(_heap[parentIdx]) >= 0)
442 {
443 break;
444 }
445
446 // Swap
447 var temp = _heap[index];
448 _heap[index] = _heap[parentIdx];
449 _heap[parentIdx] = temp;
450
451 index = parentIdx;
452 }
453 }
454
455 private void HeapifyDown(int index)
456 {
457 int lastIdx = _heap.Count - 1;
458
459 while (true)
460 {
461 int leftChild = 2 * index + 1;
462 int rightChild = 2 * index + 2;
463 int smallest = index;
464
465 if (leftChild <= lastIdx && _heap[leftChild].CompareTo(_heap[smallest]) < 0)
466 {
467 smallest = leftChild;
468 }
469
470 if (rightChild <= lastIdx && _heap[rightChild].CompareTo(_heap[smallest]) < 0)
471 {
472 smallest = rightChild;
473 }
474
475 if (smallest == index)
476 {
477 break;
478 }
479
480 // Swap
481 var temp = _heap[index];
482 _heap[index] = _heap[smallest];
483 _heap[smallest] = temp;
484
485 index = smallest;
486 }
487 }
