145 private void EnsureIndicesInitialized()
146 {
147 if (_nameIndex == null)
148 {
149 _nameIndex = new Dictionary<string, MmRoutingTableItem>();
150 _responderIndex = new Dictionary<MmResponder, MmRoutingTableItem>();
151 RebuildIndices();
152 }
153 }
154
159 private void RebuildIndices()
160 {
161 _nameIndex.Clear();
162 _responderIndex.Clear();
163
164 foreach (var item in _list)
165 {
166 if (item == null) continue;
167
168 // Name index: only store first occurrence (names may not be unique)
169 if (!string.IsNullOrEmpty(item.Name) && !_nameIndex.ContainsKey(item.Name))
170 {
171 _nameIndex[item.Name] = item;
172 }
173
174 // Responder index: responders should be unique
175 if (item.Responder != null && !_responderIndex.ContainsKey(item.Responder))
176 {
177 _responderIndex[item.Responder] = item;
178 }
179 }
180 }
181
185 private void AddToIndices(MmRoutingTableItem item)
186 {
187 if (item == null) return;
188
189 EnsureIndicesInitialized();
190
191 // Name index: only add if not already present (first occurrence wins)
192 if (!string.IsNullOrEmpty(item.Name) && !_nameIndex.ContainsKey(item.Name))
193 {
194 _nameIndex[item.Name] = item;
195 }
196
197 // Responder index: should always succeed (responders are unique)
198 if (item.Responder != null)
199 {
200 _responderIndex[item.Responder] = item;
201 }
202 }
203
207 private void RemoveFromIndices(MmRoutingTableItem item)
208 {
209 if (item == null || _nameIndex == null) return;
210
211 // Name index: only remove if this item is the indexed one
212 if (!string.IsNullOrEmpty(item.Name) &&
213 _nameIndex.TryGetValue(item.Name, out var indexedItem) &&
214 indexedItem == item)
215 {
216 _nameIndex.Remove(item.Name);
217 // Find next item with same name (if any) to re-index
218 foreach (var otherItem in _list)
219 {
220 if (otherItem != null && otherItem != item && otherItem.Name == item.Name)
221 {
222 _nameIndex[item.Name] = otherItem;
223 break;
224 }
225 }
226 }
227
228 // Responder index: simple removal
229 if (item.Responder != null)
230 {
231 _responderIndex.Remove(item.Responder);
232 }
233 }
234
238 private void InvalidateIndices()
239 {
240 _nameIndex = null;
241 _responderIndex = null;
242 }
243
248 public float CacheHitRate
249 {
250 get
251 {
252 int total = _cacheHits + _cacheMisses;
253 return total > 0 ? (float)_cacheHits / total : 0f;
254 }

260 private void EnsureCacheInitialized()
261 {
262 if (_filterCache == null)
263 {
264 _filterCache = new Dictionary<MmFilterCacheKey, List<MmRoutingTableItem>>(MAX_CACHE_SIZE);
265 _lruOrder = new LinkedList<MmFilterCacheKey>();
266 }
267 }
268
273 private void InvalidateCache()
274 {
275 if (_filterCache != null)
276 {
277 _filterCache.Clear();
278 _lruOrder.Clear();
279 }
280 }
281
287 private void TouchCacheKey(MmFilterCacheKey key)
288 {
289 // Remove from current position (if present)
290 _lruOrder.Remove(key);
291 // Add to end (most recently used)
292 _lruOrder.AddLast(key);
293 }
294
298 private void EvictLRUIfNeeded()
299 {
300 if (_filterCache.Count >= MAX_CACHE_SIZE && _lruOrder.Count > 0)
301 {
302 // Remove least recently used (first in LRU list)
303 var lruKey = _lruOrder.First.Value;
304 _lruOrder.RemoveFirst();
305 _filterCache.Remove(lruKey);
306 }
307 }

390 public List<MmRoutingTableItem> GetMmRoutingTableItems(
391 ListFilter filter = default(ListFilter),
392 MmLevelFilter levelFilter = MmLevelFilterHelper.Default)
393 {
394 // Initialize cache on first use
395 EnsureCacheInitialized();
396
397 // Create cache key from filter parameters
398 var cacheKey = new MmFilterCacheKey(filter, levelFilter);
399
400 // Check cache for existing result
401 if (_filterCache.TryGetValue(cacheKey, out var cachedResult))
402 {
403 // Cache hit - update LRU and return cached list
404 _cacheHits++;
405 TouchCacheKey(cacheKey);
406 return cachedResult;
407 }
408
409 // Cache miss - compute result by filtering routing table
410 _cacheMisses++;
411
412 // Manual filtering instead of LINQ (removed Where().ToList() for QW-5 consistency)
413 var result = new List<MmRoutingTableItem>();
414 foreach (var item in this)
415 {
416 if (CheckFilter(item, filter, levelFilter))
417 {
418 result.Add(item);
419 }
420 }
421
422 // Store in cache
423 EvictLRUIfNeeded();
424 _filterCache[cacheKey] = result;
425 TouchCacheKey(cacheKey);
426
427 return result;
428 }

477 public bool CheckFilter(MmRoutingTableItem item,
478 ListFilter listFilter, MmLevelFilter levelFilter)
479 {
480 //Level Check
481 if ((levelFilter & item.Level) == 0)
482 return false;
483
484 //List Filter check
485 if (listFilter == ListFilter.RelayNodeOnly && !(item.Responder is MmRelayNode))
486 return false;
487 if (listFilter == ListFilter.ResponderOnly && item.Responder is MmRelayNode)
488 return false;
489
490 //All conditions passed, return true
491 return true;
492 }
