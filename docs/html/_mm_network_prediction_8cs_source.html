178 public MmInputSnapshot RecordInput(Vector3 moveDirection, Vector3 rotationDelta, uint actionFlags, float deltaTime, byte[] customData = null)
179 {
180 _lastInputSequence++;
181
182 var input = new MmInputSnapshot
183 {
184 SequenceNumber = _lastInputSequence,
185 Timestamp = Time.timeAsDouble,
186 MoveDirection = moveDirection,
187 RotationDelta = rotationDelta,
188 ActionFlags = actionFlags,
189 DeltaTime = deltaTime,
190 CustomData = customData
191 };
192
193 // Apply input to predicted state
194 if (_simulationStep != null)
195 {
196 _predictedState = _simulationStep(_predictedState, input);
197 _predictedState.SequenceNumber = input.SequenceNumber;
198 }
199
200 // Store for reconciliation
201 _pendingInputs.Enqueue(input);
202
203 // Trim old inputs
204 while (_pendingInputs.Count > MaxPendingInputs)
205 {
206 _pendingInputs.Dequeue();
207 }
208
209 return input;
210 }

215 public void OnServerStateReceived(MmStateSnapshot serverState)
216 {
217 _lastServerState = serverState;
218 _lastAcknowledgedSequence = serverState.SequenceNumber;
219
220 // Remove acknowledged inputs
221 while (_pendingInputs.Count > 0 && _pendingInputs.Peek().SequenceNumber <= serverState.SequenceNumber)
222 {
223 _pendingInputs.Dequeue();
224 }
225
226 // Replay unacknowledged inputs
227 var replayState = serverState;
228 foreach (var input in _pendingInputs)
229 {
230 if (_simulationStep != null)
231 {
232 replayState = _simulationStep(replayState, input);
233 }
234 }
235
236 // Calculate error
237 _positionError = _predictedState.Position - replayState.Position;
238 _rotationError = Quaternion.Inverse(replayState.Rotation) * _predictedState.Rotation;
239
240 // Check if correction needed
241 float posError = _positionError.magnitude;
242 float rotError = Quaternion.Angle(Quaternion.identity, _rotationError);
243
244 if (posError > PositionErrorThreshold || rotError > RotationErrorThreshold)
245 {
246 // Apply smooth correction
247 _predictedState.Position = Vector3.Lerp(replayState.Position, _predictedState.Position, CorrectionSmoothing);
248 _predictedState.Rotation = Quaternion.Slerp(replayState.Rotation, _predictedState.Rotation, CorrectionSmoothing);
249 _predictedState.Velocity = replayState.Velocity;
250 _predictedState.AngularVelocity = replayState.AngularVelocity;
251 }
252 else
253 {
254 // Small error, just update state directly
255 _predictedState = replayState;
256 }
257 }

360 public MmStateSnapshot ProcessTick()
361 {
362 _currentTick++;
363
364 // Process inputs from all clients
365 foreach (var kvp in _clientInputs)
366 {
367 while (kvp.Value.Count > 0)
368 {
369 var input = kvp.Value.Dequeue();
370 if (_simulationStep != null)
371 {
372 _authoritativeState = _simulationStep(_authoritativeState, input);
373 }
374 }
375 }
376
377 // Update state metadata
378 _authoritativeState.SequenceNumber = _currentTick;
379 _authoritativeState.ServerTime = Time.timeAsDouble;
380
381 // Store in history
382 _stateHistory.Enqueue(_authoritativeState);
383 while (_stateHistory.Count > StateHistorySize)
384 {
385 _stateHistory.Dequeue();
386 }
387
388 return _authoritativeState;
389 }

430 public class MmStateInterpolator
431 {
435 public float InterpolationDelay = 0.1f;
436
440 public float MaxExtrapolation = 0.2f;
441
442 // State buffer for interpolation
443 private readonly List<MmStateSnapshot> _stateBuffer = new List<MmStateSnapshot>();
444
448 public void AddState(MmStateSnapshot state)
449 {
450 // Insert in order
451 int insertIndex = _stateBuffer.Count;
452 for (int i = 0; i < _stateBuffer.Count; i++)
453 {
454 if (_stateBuffer[i].ServerTime > state.ServerTime)
455 {
456 insertIndex = i;
457 break;
458 }
459 }
460 _stateBuffer.Insert(insertIndex, state);
461
462 // Limit buffer size
463 while (_stateBuffer.Count > 30)
464 {
465 _stateBuffer.RemoveAt(0);
466 }
467 }
468
472 public MmStateSnapshot GetInterpolatedState(double currentTime)
473 {
474 double renderTime = currentTime - InterpolationDelay;
475
476 if (_stateBuffer.Count == 0)
477 {
478 return default;
479 }
480
481 if (_stateBuffer.Count == 1)
482 {
483 return _stateBuffer[0];
484 }
485
486 // Find bracketing states
487 MmStateSnapshot before = _stateBuffer[0];
488 MmStateSnapshot after = _stateBuffer[_stateBuffer.Count - 1];
489
490 for (int i = 0; i < _stateBuffer.Count - 1; i++)
491 {
492 if (_stateBuffer[i].ServerTime <= renderTime && _stateBuffer[i + 1].ServerTime >= renderTime)
493 {
494 before = _stateBuffer[i];
495 after = _stateBuffer[i + 1];
496 break;
497 }
498 }
499
500 // Extrapolation case
501 if (renderTime > after.ServerTime)
502 {
503 float extrapolateTime = (float)(renderTime - after.ServerTime);
504 if (extrapolateTime > MaxExtrapolation)
505 {
506 extrapolateTime = MaxExtrapolation;
507 }
508 return Extrapolate(after, extrapolateTime);
509 }
510
511 // Interpolation
512 float t = (float)((renderTime - before.ServerTime) / (after.ServerTime - before.ServerTime));
513 t = Mathf.Clamp01(t);
514 return Interpolate(before, after, t);
515 }
516
517 private MmStateSnapshot Interpolate(MmStateSnapshot a, MmStateSnapshot b, float t)
518 {
519 return new MmStateSnapshot
520 {
521 SequenceNumber = t < 0.5f ? a.SequenceNumber : b.SequenceNumber,
522 ServerTime = Mathf.Lerp((float)a.ServerTime, (float)b.ServerTime, t),
523 Position = Vector3.Lerp(a.Position, b.Position, t),
524 Rotation = Quaternion.Slerp(a.Rotation, b.Rotation, t),
525 Velocity = Vector3.Lerp(a.Velocity, b.Velocity, t),
526 AngularVelocity = Vector3.Lerp(a.AngularVelocity, b.AngularVelocity, t)
527 };
528 }
529
530 private MmStateSnapshot Extrapolate(MmStateSnapshot state, float deltaTime)
531 {
532 return new MmStateSnapshot
533 {
534 SequenceNumber = state.SequenceNumber,
535 ServerTime = state.ServerTime + deltaTime,
536 Position = state.Position + state.Velocity * deltaTime,
537 Rotation = state.Rotation * Quaternion.Euler(state.AngularVelocity * deltaTime),
538 Velocity = state.Velocity,
539 AngularVelocity = state.AngularVelocity
540 };
541 }
542
546 public void Clear()
547 {
548 _stateBuffer.Clear();
549 }
550 }
