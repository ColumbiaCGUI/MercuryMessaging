102 public static class MmPathSpecification
103 {
104 // Cache for parsed paths (performance optimization)
105 private static Dictionary<string, ParsedPath> pathCache = new Dictionary<string, ParsedPath>();
106
107 // Maximum cache size to prevent unbounded growth
108 private const int MaxCacheSize = 100;
109
117 public static ParsedPath Parse(string path)
118 {
119 // Validate input
120 if (string.IsNullOrEmpty(path))
121 {
122 throw new MmInvalidPathException("Path cannot be null or empty");
123 }
124
125 // Check cache first
126 if (pathCache.TryGetValue(path, out ParsedPath cached))
127 {
128 return cached;
129 }
130
131 // Parse the path
132 ParsedPath parsedPath = ParseInternal(path);
133
134 // Add to cache (with size limit)
135 if (pathCache.Count < MaxCacheSize)
136 {
137 pathCache[path] = parsedPath;
138 }
139 else
140 {
141 // Cache full - clear and restart (simple eviction strategy)
142 pathCache.Clear();
143 pathCache[path] = parsedPath;
144 }
145
146 return parsedPath;
147 }
148
152 private static ParsedPath ParseInternal(string path)
153 {
154 // Remove leading/trailing whitespace
155 path = path.Trim();
156
157 // Check for trailing slash
158 if (path.EndsWith("/"))
159 {
160 throw new MmInvalidPathException(
161 $"Path cannot end with '/': {path}"
162 );
163 }
164
165 // Split by separator
166 string[] parts = path.Split('/');
167
168 // Validate and convert each part
169 List<PathSegment> segments = new List<PathSegment>();
170
171 for (int i = 0; i < parts.Length; i++)
172 {
173 string part = parts[i].Trim();
174
175 // Empty segment (double slash or leading slash)
176 if (string.IsNullOrEmpty(part))
177 {
178 throw new MmInvalidPathException(
179 $"Empty segment at position {i} in path: {path}"
180 );
181 }
182
183 // Parse segment
184 PathSegment segment;
185 if (!TryParseSegment(part, out segment))
186 {
187 throw new MmInvalidPathException(
188 $"Invalid segment '{part}' at position {i}. " +
189 "Valid segments: parent, child, sibling, self, ancestor, descendant, *"
190 );
191 }
192
193 segments.Add(segment);
194 }
195
196 // Validate path structure
197 ValidatePath(segments, path);
198
199 return new ParsedPath(path, segments.ToArray());
200 }
201
205 private static bool TryParseSegment(string segmentStr, out PathSegment segment)
206 {
207 switch (segmentStr.ToLowerInvariant())
208 {
209 case "parent":
210 segment = PathSegment.Parent;
211 return true;
212 case "child":
213 segment = PathSegment.Child;
214 return true;
215 case "sibling":
216 segment = PathSegment.Sibling;
217 return true;
218 case "self":
219 segment = PathSegment.Self;
220 return true;
221 case "ancestor":
222 segment = PathSegment.Ancestor;
223 return true;
224 case "descendant":
225 segment = PathSegment.Descendant;
226 return true;
227 case "*":
228 segment = PathSegment.Wildcard;
229 return true;
230 default:
231 segment = PathSegment.Self; // Default (not used)
232 return false;
233 }
234 }
235
239 private static void ValidatePath(List<PathSegment> segments, string originalPath)
240 {
241 if (segments.Count == 0)
242 {
243 throw new MmInvalidPathException("Path must contain at least one segment");
244 }
245
246 // Check for invalid wildcard usage
247 for (int i = 0; i < segments.Count; i++)
248 {
249 if (segments[i] == PathSegment.Wildcard)
250 {
251 // Wildcard cannot be first segment (no context)
252 if (i == 0)
253 {
254 throw new MmInvalidPathException(
255 $"Wildcard '*' cannot be the first segment in path: {originalPath}. " +
256 "Wildcard requires a preceding relationship to expand."
257 );
258 }
259
260 // Wildcard must be followed by another segment
261 if (i == segments.Count - 1)
262 {
263 throw new MmInvalidPathException(
264 $"Wildcard '*' cannot be the last segment in path: {originalPath}. " +
265 "Wildcard must be followed by a relationship to navigate."
266 );
267 }
268
269 // Wildcard cannot be followed by another wildcard
270 if (i + 1 < segments.Count && segments[i + 1] == PathSegment.Wildcard)
271 {
272 throw new MmInvalidPathException(
273 $"Consecutive wildcards '**' not supported in path: {originalPath}. " +
274 "Use explicit segments between wildcards."
275 );
276 }
277 }
278 }
