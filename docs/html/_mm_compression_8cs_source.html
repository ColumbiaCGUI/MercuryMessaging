111 public const byte CompressedPacketMarker = 0xC0;
112
116 public const byte UncompressedPacketMarker = 0x00;

122 public static byte[] Compress(byte[] data, MmCompressionConfig config)
123 {
124 if (config == null) config = MmCompressionConfig.Default;
125
126 // Skip compression for small payloads or if disabled
127 if (config.Type == MmCompressionType.None || data.Length < config.MinSizeToCompress)
128 {
129 return WrapUncompressed(data);
130 }
131
132 byte[] compressed = CompressInternal(data, config);
133
134 // Check if compression is beneficial
135 if (compressed.Length >= data.Length * config.MinCompressionRatio)
136 {
137 return WrapUncompressed(data);
138 }
139
140 return WrapCompressed(compressed, config.Type, data.Length);
141 }
142
146 public static byte[] Decompress(byte[] data)
147 {
148 if (data == null || data.Length < 2)
149 {
150 throw new ArgumentException("Invalid compressed data");
151 }
152
153 byte marker = data[0];
154
155 if (marker == UncompressedPacketMarker)
156 {
157 // Uncompressed - just unwrap
158 byte[] result = new byte[data.Length - 1];
159 Array.Copy(data, 1, result, 0, result.Length);
160 return result;
161 }
162 else if (marker == CompressedPacketMarker)
163 {
164 // Compressed - read header and decompress
165 if (data.Length < 6)
166 {
167 throw new ArgumentException("Invalid compressed packet header");
168 }
169
170 MmCompressionType type = (MmCompressionType)data[1];
171 int originalLength = BitConverter.ToInt32(data, 2);
172
173 byte[] compressedData = new byte[data.Length - 6];
174 Array.Copy(data, 6, compressedData, 0, compressedData.Length);
175
176 return DecompressInternal(compressedData, type, originalLength);
177 }
178 else
179 {
180 throw new ArgumentException($"Unknown compression marker: 0x{marker:X2}");
181 }
182 }
183
187 public static bool IsCompressed(byte[] data)
188 {
189 return data != null && data.Length > 0 && data[0] == CompressedPacketMarker;
190 }
191
195 public static float GetCompressionRatio(byte[] original, byte[] compressed)
196 {
197 if (original == null || original.Length == 0) return 1.0f;
198 return (float)compressed.Length / original.Length;
199 }
200
201 private static byte[] WrapUncompressed(byte[] data)
202 {
203 byte[] result = new byte[data.Length + 1];
204 result[0] = UncompressedPacketMarker;
205 Array.Copy(data, 0, result, 1, data.Length);
206 return result;
207 }
208
209 private static byte[] WrapCompressed(byte[] compressed, MmCompressionType type, int originalLength)
210 {
211 byte[] result = new byte[compressed.Length + 6];
212 result[0] = CompressedPacketMarker;
213 result[1] = (byte)type;
214 byte[] lengthBytes = BitConverter.GetBytes(originalLength);
215 Array.Copy(lengthBytes, 0, result, 2, 4);
216 Array.Copy(compressed, 0, result, 6, compressed.Length);
217 return result;
218 }
219
220 private static byte[] CompressInternal(byte[] data, MmCompressionConfig config)
221 {
222 CompressionLevel level = config.Level switch
223 {
224 MmCompressionLevel.Fastest => System.IO.Compression.CompressionLevel.Fastest,
225 MmCompressionLevel.Optimal => System.IO.Compression.CompressionLevel.Optimal,
226 _ => System.IO.Compression.CompressionLevel.NoCompression
227 };
228
229 using (var outputStream = new MemoryStream())
230 {
231 Stream compressionStream = config.Type switch
232 {
233 MmCompressionType.GZip => new GZipStream(outputStream, level, leaveOpen: true),
234 MmCompressionType.Deflate => new DeflateStream(outputStream, level, leaveOpen: true),
235 _ => throw new ArgumentException($"Unsupported compression type: {config.Type}")
236 };
237
238 using (compressionStream)
239 {
240 compressionStream.Write(data, 0, data.Length);
241 }
242
243 return outputStream.ToArray();
244 }
245 }
246
247 private static byte[] DecompressInternal(byte[] data, MmCompressionType type, int originalLength)
248 {
249 using (var inputStream = new MemoryStream(data))
250 {
251 Stream decompressionStream = type switch
252 {
253 MmCompressionType.GZip => new GZipStream(inputStream, CompressionMode.Decompress),
254 MmCompressionType.Deflate => new DeflateStream(inputStream, CompressionMode.Decompress),
255 _ => throw new ArgumentException($"Unsupported compression type: {type}")
256 };
257
258 using (decompressionStream)
259 using (var outputStream = new MemoryStream(originalLength))
260 {
261 byte[] buffer = new byte[4096];
262 int bytesRead;
263 while ((bytesRead = decompressionStream.Read(buffer, 0, buffer.Length)) > 0)
264 {
265 outputStream.Write(buffer, 0, bytesRead);
266 }
267 return outputStream.ToArray();
268 }
269 }
270 }

359 if (bytes < 1024) return $"{bytes}B";
