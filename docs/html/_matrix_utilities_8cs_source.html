117 public static Matrix SubtractMatrix(this Matrix lhs, Matrix rhs)
118 {
119 Matrix result = Matrix.identity;
120
121 result.SetColumn(0, lhs.GetColumn(0) - rhs.GetColumn(0));
122 result.SetColumn(1, lhs.GetColumn(1) - rhs.GetColumn(1));
123 result.SetColumn(2, lhs.GetColumn(2) - rhs.GetColumn(2));
124 result.SetColumn(3, lhs.GetColumn(3) - rhs.GetColumn(3));
125
126 return result;
127 }

135 public static Matrix AddMatrix(this Matrix lhs, Matrix rhs)
136 {
137 Matrix result = Matrix.identity;
138
139 result.SetColumn(0, lhs.GetColumn(0) + rhs.GetColumn(0));
140 result.SetColumn(1, lhs.GetColumn(1) + rhs.GetColumn(1));
141 result.SetColumn(2, lhs.GetColumn(2) + rhs.GetColumn(2));
142 result.SetColumn(3, lhs.GetColumn(3) + rhs.GetColumn(3));
143
