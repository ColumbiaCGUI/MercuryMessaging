110 IEnumerator ShowingMessage(float displayTime)
111 {
112 var canvas = TopRightPanel.GetComponent<CanvasGroup>();
113
114 msgStartingAlpha = canvas.alpha;
115
116 yield return new WaitForSeconds(displayTime);
117
118 while (canvas.alpha > 0)
119 {
120 canvas.alpha -= Time.deltaTime / MessageFadeTime;
121 yield return null;
122 }
123
124 TopRightPanel.SetActive(false);
125
126 canvas.alpha = msgStartingAlpha;
127
128 yield return null;
129 }
