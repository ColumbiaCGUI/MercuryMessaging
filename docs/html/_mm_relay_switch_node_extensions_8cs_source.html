42 public static FsmConfigBuilder<MmRoutingTableItem> ConfigureStates(this MmRelaySwitchNode switchNode)
43 {
44 if (switchNode == null)
45 throw new ArgumentNullException(nameof(switchNode));
46
47 if (switchNode.RespondersFSM == null)
48 throw new InvalidOperationException("RespondersFSM is not initialized. Ensure Awake() has been called.");
49
50 // Create resolver that looks up states by name in the routing table
51 Func<string, MmRoutingTableItem> stateResolver = stateName =>
52 {
53 try
54 {
55 return switchNode.RoutingTable[stateName];
56 }
57 catch
58 {
59 UnityEngine.Debug.LogWarning($"[MmRelaySwitchNode] State '{stateName}' not found in routing table.");
60 return null;
61 }
62 };
63
64 return new FsmConfigBuilder<MmRoutingTableItem>(switchNode.RespondersFSM, stateResolver);
65 }
