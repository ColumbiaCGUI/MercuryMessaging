87 public bool TryGetNetworkId(GameObject gameObject, out uint networkId)
88 {
89#if FUSION2_AVAILABLE
90 if (gameObject != null)
91 {
92 var nob = gameObject.GetComponent<NetworkObject>();
93 // Check IsValid and that Raw is not 0 (invalid/unspawned)
94 if (nob != null && nob.IsValid && nob.Id.Raw > 0)
95 {
96 networkId = nob.Id.Raw;
97 return true;
98 }
99 }
100#endif
101 networkId = 0;
102 return false;
103 }

106 public bool TryGetGameObject(uint networkId, out GameObject gameObject)
107 {
108#if FUSION2_AVAILABLE
109 var runner = Runner;
110 if (runner != null && runner.IsRunning && networkId > 0)
111 {
112 // Create NetworkId from raw value
113 var netId = new NetworkId { Raw = networkId };
114
115 // Try to find the object in the runner's registry
116 if (runner.TryFindObject(netId, out NetworkObject nob))
117 {
118 gameObject = nob.gameObject;
119 return true;
120 }
121 }
122#endif
123 gameObject = null;
124 return false;
125 }
