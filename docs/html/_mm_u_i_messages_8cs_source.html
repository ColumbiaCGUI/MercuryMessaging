93 public class MmUIClickMessage : MmMessage
94 {
96 public Vector2 Position;
97
99 public int ClickCount;
100
102 public int Button;
103
105 public bool IsDoubleClick => ClickCount >= 2;
106
108 public bool IsRightClick => Button == 1;
109
110 public MmUIClickMessage() : base((MmMethod)MmUIMethod.Click, (MmMessageType)MmUIMessageType.UIClick)
111 {
112 }
113
114 public MmUIClickMessage(Vector2 position, int clickCount = 1, int button = 0, MmMetadataBlock metadataBlock = null)
115 : base((MmMethod)MmUIMethod.Click, (MmMessageType)MmUIMessageType.UIClick, metadataBlock)
116 {
117 Position = position;
118 ClickCount = clickCount;
119 Button = button;
120 }
121
122 public MmUIClickMessage(MmUIClickMessage message) : base(message)
123 {
124 Position = message.Position;
125 ClickCount = message.ClickCount;
126 Button = message.Button;
127 }
128
129 public override MmMessage Copy()
130 {
131 return new MmUIClickMessage(this);
132 }
133
134 public override int Deserialize(object[] data)
135 {
136 int index = base.Deserialize(data);
137 Position = new Vector2((float)data[index++], (float)data[index++]);
138 ClickCount = (int)data[index++];
139 Button = (int)data[index++];
140 return index;
141 }
142
143 public override object[] Serialize()
144 {
145 return base.Serialize().Concat(new object[] { Position.x, Position.y, ClickCount, Button }).ToArray();
146 }
147 }
