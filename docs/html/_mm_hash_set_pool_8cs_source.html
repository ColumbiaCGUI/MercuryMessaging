33using System.Collections.Generic;
34using UnityEngine.Pool;
35
36namespace MercuryMessaging
37{
47 public static class MmHashSetPool
48 {
49 #region Configuration
50
54 public const int DEFAULT_CAPACITY = 100;
55
59 public const int MAX_SIZE = 1000;
60
64 private const int HASHSET_INITIAL_CAPACITY = 16;
65
69 #if UNITY_EDITOR
70 private const bool COLLECTION_CHECK = true;
71 #else
72 private const bool COLLECTION_CHECK = false;
73 #endif
74
75 #endregion
76
77 #region Pool
78
79 private static readonly ObjectPool<HashSet<int>> _pool = new ObjectPool<HashSet<int>>(
80 createFunc: () => new HashSet<int>(HASHSET_INITIAL_CAPACITY),
81 actionOnGet: set => set.Clear(), // Always start with empty set
82 actionOnRelease: null,
83 actionOnDestroy: null,
84 collectionCheck: COLLECTION_CHECK,
85 defaultCapacity: DEFAULT_CAPACITY,
86 maxSize: MAX_SIZE
87 );
88
89 #endregion
90
91 #region Public API
92
98 public static HashSet<int> Get()
99 {
100 return _pool.Get();
101 }
102
107 public static void Return(HashSet<int> set)
108 {
109 if (set != null)
110 {
111 _pool.Release(set);
112 }
113 }
114
121 public static HashSet<int> GetCopy(HashSet<int> source)
122 {
123 var set = _pool.Get();
124 if (source != null)
125 {
126 foreach (var id in source)
127 {
128 set.Add(id);
129 }
130 }
131 return set;
132 }
133
134 #endregion
135
136 #region Statistics (Editor Only)
137
138 #if UNITY_EDITOR
142 public static string GetStatistics()
143 {
144 return $"MmHashSetPool: {_pool.CountActive} active, {_pool.CountInactive} inactive";
145 }
146
150 public static int ActiveCount => _pool.CountActive;
151
155 public static int InactiveCount => _pool.CountInactive;
156 #endif
157
158 #endregion
159 }
