63 public override void MmInvoke (MmMessage message)
64 {
65 switch (message.MmMethod)
66 {
67 case MmMethod.MessageTransform:
68 MmMessageTransform newMsg = message as MmMessageTransform;
69 HandleTransform (newMsg);
70 break;
71 default:
72 base.MmInvoke(message);
73 break;
74 }
75 }

77 public override void Update ()
78 {
79 if (CanSend && Time.time - lastClientSendTime > (1/NetworkSendInterval))
80 {
81 lastClientSendTime = Time.time;
82
83 GetRelayNode()
84 .MmInvoke(MmMethod.MessageTransform,
85 new MmMessageTransform(
86 new MmTransform(gameObject.transform, useGlobalTransformation),
87 !useGlobalTransformation),
88 MmMessageType.MmTransform,
89 new MmMetadataBlock(MmLevelFilter.Self,
90 MmActiveFilter.All, MmSelectedFilter.All)
91 );
92 }
93
94 base.Update();
95 }

99 //Debug.Log(gameObject.name + " HandleTransform");
