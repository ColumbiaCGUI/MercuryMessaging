365 #region Execution
366
370 [MethodImpl(MethodImplOptions.AggressiveInlining)]
371 public void Execute()
372 {
373 // FAIL-FAST: Log error if no relay node instead of silently failing
374 if (_relay == null)
375 {
376 UnityEngine.Debug.LogError("[MmDeferredRoutingBuilder] Execute called with no relay node. Message will not be sent.");
377 return;
378 }
379
380 // Nothing to execute if no payload was set
381 if (!_hasPayload)
382 {
383 UnityEngine.Debug.LogWarning("[MmDeferredRoutingBuilder] Execute called but no payload was set. Call Send() before Execute().");
384 return;
385 }
386
387 // Delegate to MmFluentMessage for the actual routing logic
388 var fluent = new MmFluentMessage(_relay, _method, _payload);
389 fluent = fluent.To(_levelFilter);
390
391 // Apply filters
392 if (_activeFilter == MmActiveFilter.Active)
393 fluent = fluent.Active();
394 if (_selectedFilter == MmSelectedFilter.Selected)
395 fluent = fluent.Selected();
396 if (_networkFilter == MmNetworkFilter.All)
397 fluent = fluent.OverNetwork();
398 else if (_networkFilter == MmNetworkFilter.Network)
399 fluent = fluent.NetworkOnly();
400 if (_tag != MmTagHelper.Everything)
401 fluent = fluent.WithTag(_tag);
402
403 fluent.Execute();
404 }
405
406 #endregion
