222 public string Validate()
223 {
224 if (HistoryCacheSizeMs < 0)
225 return "HistoryCacheSizeMs must be >= 0";
226
227 if (MaxRoutingHops < -1)
228 return "MaxRoutingHops must be >= -1 (-1 = unlimited)";
229
230 if (MaxRoutingHops == 0)
231 return "MaxRoutingHops cannot be 0 (use -1 for unlimited or >= 1 for limit)";
232
233 if (ProfilingThresholdMs < 0)
234 return "ProfilingThresholdMs must be >= 0";
235
236 // Warning: unlimited hops with lateral routing can cause infinite loops
237 if (MaxRoutingHops == -1 && AllowLateralRouting && !EnableHistoryTracking)
238 {
239 return "Warning: Unlimited hops with lateral routing requires history tracking to prevent infinite loops";
240 }
241
242 return null; // Valid
243 }
