200 public virtual void ProceedToNextTask()
201 {
202 if(SavePartialProgression)
203 TaskInfoCollectionLoader.SaveCurrentTaskSequenceValue(CurrentTaskInfo.UserSequence);
204
205 currentTaskInfo = (currentTaskInfo == null)
206 ? TaskInfos.First
207 : currentTaskInfo.Next;
208
209 if (ShouldTriggerSwitch())
210 {
211 TasksNode.MmInvoke(MmMethod.Switch, CurrentTaskInfo.TaskName,
212 new MmMetadataBlock(MmLevelFilter.Self, MmActiveFilter.All));
213 }

232 public LinkedListNode<U> GetNodeAt(int index)
233 {
234 if (index >= TaskInfos.Count) return null;
235
236 var count = 0;
237 var task = TaskInfos.First;
238
239 while (task != null)
240 {
241 if (index == count) return task;
242 task = task.Next;
243 count++;
244 }
245
246 return null;
247 }
