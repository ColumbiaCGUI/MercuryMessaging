136 public DataCollectionBuilder Collect(string name, Func<string> dataProvider)
137 {
138 if (string.IsNullOrEmpty(name))

242 public MmDataCollector Build()
243 {
244 // Apply settings
245 _collector.OutputType = _outputType;
246 _collector.MmDataCollectionFreq = _frequency;
247
248 // Clear and add items
249 _collector.Clear();
250 foreach (var (name, printData) in _dataItems)
251 {
252 _collector.Add(name, printData);
253 }
254
255 return _collector;
256 }
257
261 public MmDataCollector Start()
262 {
263 Build();
264
265 // Create handler and start
266 _collector.CreateDataHandler(_fileLocation, _fileName);
267 _collector.OpenTag();
268
269 return _collector;
270 }
