192 private void AutoRegisterRelayNodes()
193 {
194 // Find all MmRelayNodes in scene, INCLUDING inactive ones
195 var relayNodes = FindObjectsByType<MmRelayNode>(FindObjectsInactive.Include, FindObjectsSortMode.None);
196 int registeredWithNetId = 0;
197 int registeredWithFallback = 0;
198
199 foreach (var node in relayNodes)
200 {
201 if (_resolver.TryGetNetworkId(node.gameObject, out uint netId))
202 {
203 // Has valid NetworkObject with ID
204 _bridge.RegisterRelayNode(netId, node);
205 registeredWithNetId++;
206 Log($"Registered relay node '{node.gameObject.name}' with NetworkObject ID: {netId}");
207 }
208 else
209 {
210 // Fallback: Use deterministic hash based on hierarchy path
211 uint fallbackId = GetDeterministicId(node.gameObject);
212 _bridge.RegisterRelayNode(fallbackId, node);
213 registeredWithFallback++;
214 Log($"Registered relay node '{node.gameObject.name}' with fallback ID: {fallbackId} (path-based)");
215 }
216 }
217
218 Log($"Auto-registered {registeredWithNetId} relay nodes with NetworkObject IDs, {registeredWithFallback} with fallback IDs");
219 }

224 private uint GetDeterministicId(GameObject go)
225 {
226 string path = GetHierarchyPath(go);
227 int hash = path.GetHashCode();
228 return (uint)((hash & 0x7FFFFFFF) | 1);
229 }
230
231 private string GetHierarchyPath(GameObject go)
232 {
233 var path = go.name;
234 var parent = go.transform.parent;
235 while (parent != null)
236 {
237 path = parent.name + "/" + path;
238 parent = parent.parent;
239 }
240 return path;
241 }
