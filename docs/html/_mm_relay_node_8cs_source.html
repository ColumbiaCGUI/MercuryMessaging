376 [MethodImpl(MethodImplOptions.NoInlining)]
377 public void UpdateMessages(MmMessage message)
378 {
379 List<MmRoutingTableItem> items = new List<MmRoutingTableItem>();
380
381 // Determine which level to filter by
382 if (message.MetadataBlock.LevelFilter == MmLevelFilter.Parent)
383 {
384 items = RoutingTable.GetMmRoutingTableItems(MmRoutingTable.ListFilter.All, MmLevelFilter.Parent);
385 }
386 else if (message.MetadataBlock.LevelFilter == MmLevelFilter.Child)
387 {
388 items = RoutingTable.GetMmRoutingTableItems(MmRoutingTable.ListFilter.All, MmLevelFilter.Child);
389 }
390
391 // Update items and propagate the message recursively
392 foreach (MmRoutingTableItem item in items)
393 {
394 if (item.Tags == message.MetadataBlock.Tag || message.MetadataBlock.Tag == (MmTag)(-1))
395 {
396 UpdateItemAndPropagate(item, message);
397 }
398 }
399 }

405 [MethodImpl(MethodImplOptions.NoInlining)]
406 private void UpdateItemAndPropagate(MmRoutingTableItem item, MmMessage message)
407 {
408 System.DateTime currentTime = System.DateTime.Now;
409 // Update the messageOutList for the current node (null check for inactive GameObjects)
410 messageOutList?.Insert(0, item.Name + " : " + message.MmMessageType.ToString() + "\n" + currentTime.ToString("yyyy-MM-dd HH:mm:ss"));
411 // No truncation needed - circular buffer handles it automatically
412
413 // If the item has a responder that is a MmRelayNode, update its messageInList
414 if (item.Responder is MmRelayNode relayNode && item.Responder != this)
415 {
416 // Null check for messageInList - may be null if Awake() hasn't been called (inactive GameObject)
417 relayNode.messageInList?.Insert(0, gameObject.name + " : " + message.MmMessageType.ToString() + "\n" + currentTime.ToString("yyyy-MM-dd HH:mm:ss"));
418 // No truncation needed - circular buffer handles it automatically
419
420 // Recursively update the child nodes of the current relay node
421 relayNode.UpdateMessages(message);
422 }

471 public virtual MmRoutingTableItem MmAddToRoutingTable(MmResponder mmResponder, MmLevelFilter level)
472 {
473 bool tagCheckEnabled = mmResponder.TagCheckEnabled;
474 var routingTableItem = new MmRoutingTableItem (mmResponder.name, mmResponder) {
475 Level = level,
476 Tags = mmResponder.Tag, // Copy tag from responder to routing table item
477 TagCheckEnabled = tagCheckEnabled // Phase 8: Cache for removal tracking
478 };
479
480 if (RoutingTable.Contains(mmResponder))
481 return null; // Already in list
482
483 // Phase 8: Track tag-check-enabled responders for fast-path optimization
484 if (tagCheckEnabled)
485 {
486 _tagCheckEnabledCount++;
487 }
488
489 //If there is an MmInvoke executing, add it to the
490 // MmRespondersToAdd queue.
491 if (doNotModifyRoutingTable)
492 {
493 MmRespondersToAdd.Enqueue(routingTableItem);
494 }
495 else
496 {
497 RoutingTable.Add(routingTableItem);
498 }
499
500 return routingTableItem;
501 }

578 public void MmRefreshResponders()
579 {
580 // Get all MmResponders and filter out MmRelayNodes (removed LINQ for performance)
581 var components = GetComponents<MmResponder>();
582 List<MmResponder> responders = new List<MmResponder>(components.Length);
583 foreach (var component in components)
584 {
585 if (!(component is MmRelayNode))
586 {
587 responders.Add(component);
588 }
589 }
590
591 // Add own implementations of IMmResponder to priority list
592 // Also update Tags for existing responders (in case tags changed)
593 foreach (var responder in responders)
594 {
595 if (!RoutingTable.Contains(responder))
596 {
597 MmAddToRoutingTable(responder, MmLevelFilter.Self);
598 }
599 else
600 {
601 // Update Tags for existing routing table item
602 var existingItem = RoutingTable[responder];
603 if (existingItem != null)
604 {
605 existingItem.Tags = responder.Tag;
606 }
607 }
608 }
609
610 for (int i = RoutingTable.Count - 1; i >= 0; i--)
611 {
612 var item = RoutingTable[i];
613 if (item.Responder == null)
614 {
615 // Phase 8: Decrement tag counter using cached value
616 if (item.TagCheckEnabled)
617 {
618 _tagCheckEnabledCount--;
619 }
620 RoutingTable.RemoveAt(i);
621 }
622 }

629 public void RefreshParents()
630 {
631 MmLogger.LogFramework("Refreshing parents on MmRelayNode: " + gameObject.name);
632
633 // Iterate through routing table and process child nodes (removed LINQ for performance)
634 foreach (var child in RoutingTable)
635 {
636 if (child.Level == MmLevelFilter.Child)
637 {
638 // Only process if the child responder has its own relay node
639 // Plain responders (like NetworkTestResponder) don't have relay nodes
640 var childNode = child.Responder?.GetRelayNode();
641 if (childNode != null && childNode != this)
642 {
643 childNode.AddParent(this);
644 childNode.RefreshParents();
645 }
646 }
647 }
648
649 // Update parent entries in routing table (removed LINQ for performance)
650 foreach (var parent in MmParentList)
651 {
652 // Find parent in routing table (manual search avoids LINQ allocation)
653 MmRoutingTableItem foundItem = null;
654 foreach (var item in RoutingTable)
655 {
656 if (item.Responder == parent)
657 {
658 foundItem = item;
659 break;
660 }
661 }
662
663 if (foundItem == null) {
664 MmAddToRoutingTable (parent, MmLevelFilter.Parent);
665 }
666 else
667 {
668 foundItem.Level = MmLevelFilter.Parent;
669 }
670 }

731 internal void RemoveListener<T>(MmListenerSubscription<T> subscription) where T : MmMessage
732 {
733 if (_listeners == null || subscription == null) return;
734
735 // Find and remove the wrapper
736 for (int i = _listeners.Count - 1; i >= 0; i--)
737 {
738 if (_listeners[i] is MmListenerSubscriptionImpl<T> wrapper &&
739 wrapper.GetInner() == subscription)
740 {
741 _listeners.RemoveAt(i);
742 break;
743 }
744 }
745 }

754 // Dispose all listeners

767 private void NotifyListeners(MmMessage message)
768 {
769 if (_listeners == null || _listeners.Count == 0) return;
770
771 // Iterate backwards to allow removal during iteration
772 for (int i = _listeners.Count - 1; i >= 0; i--)
773 {
774 var listener = _listeners[i];
775 if (listener.IsDisposed)
776 {
777 _listeners.RemoveAt(i);
778 continue;
779 }
780
781 // TryInvoke returns false if the listener should be removed (e.g., one-time listeners)
782 if (!listener.TryInvoke(message))
783 {
784 _listeners.RemoveAt(i);
785 }
786 }

805 public override void MmInvoke(MmMessage message)
806 {
807 // Debug tracking - disabled in PerformanceMode
808 if (!PerformanceMode)
809 {
810 UpdateMessages(message);
811 }
812
813 // DSL Phase 2.1: Notify listeners before responder dispatch
814 // This allows external components to subscribe to messages without creating responders
815 NotifyListeners(message);
816
817 MmMessageType msgType = message.MmMessageType;
818 //If the MmRelayNode has not been initialized, initialize it here,
819 // and refresh the parents - to ensure proper routing can occur.
820 InitializeNode();
821
822 // Check hop limit to prevent infinite loops
823 if (maxMessageHops > 0)
824 {
825 if (message.HopCount >= maxMessageHops)
826 {
827 string errorMsg = $"[HOP LIMIT] Message dropped at '{name}' after {message.HopCount} hops (max: {maxMessageHops}). Method: {message.MmMethod}";
828 if (StrictMode)
829 {
830 throw new System.InvalidOperationException(errorMsg);
831 }
832 MmLogger.LogFramework(errorMsg);
833 return;
834 }
835
836 // Increment hop counter for this relay
837 message.HopCount++;
838 }
839
840 // Check for cycles if cycle detection is enabled
841 if (enableCycleDetection)
842 {
843 int nodeInstanceId = gameObject.GetInstanceID();
844
845 // Initialize visited nodes set if not already done (use pool for performance)
846 if (message.VisitedNodes == null)
847 {
848 message.VisitedNodes = MmHashSetPool.Get();
849 }
850
851 // Check if we've already visited this node (circular path detected)
852 if (message.VisitedNodes.Contains(nodeInstanceId))
853 {
854 string errorMsg = $"[CYCLE DETECTED] Message dropped at '{name}' - circular path detected. Method: {message.MmMethod}, Hops: {message.HopCount}";
855 if (StrictMode)
856 {
857 throw new System.InvalidOperationException(errorMsg);
858 }
859 MmLogger.LogFramework(errorMsg);
860 return;
861 }
862
863 // Mark this node as visited
864 message.VisitedNodes.Add(nodeInstanceId);
865 }
866
867 doNotModifyRoutingTable = true;
868 MmNetworkFilter networkFilter = message.MetadataBlock.NetworkFilter;
869
870 // If an MmNetworkResponder is attached to this object, and the MmMessage has not already been deserialized
871 // then call the MmNetworkResponder's network message invocation function.
872 if (MmNetworkResponder != null &&
873 message.MetadataBlock.NetworkFilter != MmNetworkFilter.Local &&
874 !message.IsDeserialized)
875 {
876 //if (!dirty)
877 //{
878 // dirty = true;
879 // message.TimeStamp = DateTime.UtcNow.ToShortTimeString();
880 // _prevMessageTime = message.TimeStamp;
881 //}
882
883 //This will ensure that beyond the point at which a message is determined to be sendable,
884 // it will not be treated as networ
885 networkFilter = NetworkFilterAdjust(ref message);
886
887 MmNetworkResponder.MmInvoke (message);
888 }
889
890
891 //Lazy message copying: Only create copies if actually needed for multiple directions
892 //This reduces 20-30% overhead from unnecessary message allocations
893
894 MmLevelFilter levelFilter = message.MetadataBlock.LevelFilter;
895 MmActiveFilter activeFilter = ActiveFilterAdjust(ref message);
896 MmSelectedFilter selectedFilter = SelectedFilterAdjust(ref message);
897
898 //If this message was a network-only message and
899 // this node does not allow for propagation of network messages,
900 // then return.
901 if (!AllowNetworkPropagationLocally && !message.IsDeserialized &&
902 message.MetadataBlock.NetworkFilter == MmNetworkFilter.Network)
903 {
904 return;
905 }
906
907 // CRITICAL: Check if advanced routing filters are present
908 // If yes, standard routing should only deliver to Self responders
909 // (Advanced routing will handle Parent/Child/Descendant/Ancestor/Sibling/Cousin deliveries)
910 bool hasAdvancedFilters = (levelFilter & (MmLevelFilter.Parent | MmLevelFilter.Descendants |
911 MmLevelFilter.Ancestors | MmLevelFilter.Siblings |
912 MmLevelFilter.Cousins | MmLevelFilter.Custom)) != 0;
913
914 // First pass: determine which directions are needed
915 bool needsParent = false;
916 bool needsChild = false;
917 bool needsSelf = false;
918
919 foreach (var routingTableItem in RoutingTable)
920 {
921 MmLevelFilter responderLevel = routingTableItem.Level;
922
923 if ((responderLevel & MmLevelFilter.Parent) > 0)
924 needsParent = true;
925 else if ((responderLevel & MmLevelFilter.Child) > 0)
926 needsChild = true;
927 else
928 needsSelf = true;
929 }
930
931 // Create messages lazily based on what's needed
932 MmMessage upwardMessage = null;
933 MmMessage downwardMessage = null;
934
935 // If we need multiple directions, we need to copy
936 int directionsNeeded = (needsParent ? 1 : 0) + (needsChild ? 1 : 0) + (needsSelf ? 1 : 0);
937
938 if (directionsNeeded > 1)
939 {
940 // Need copies for multiple directions
941 if (needsParent)
942 {
943 upwardMessage = message.Copy();
944 upwardMessage.MetadataBlock.LevelFilter = MmLevelFilterHelper.SelfAndParents;
945 }
946
947 if (needsChild)
948 {
949 downwardMessage = message.Copy();
950 downwardMessage.MetadataBlock.LevelFilter = MmLevelFilterHelper.SelfAndChildren;
951 }
952
953 // needsSelf uses original message
954 }
955 else if (directionsNeeded == 1)
956 {
957 if (hasAdvancedFilters)
958 {
959 // Advanced routing active - create copies to preserve original message
960 if (needsParent)
961 {
962 upwardMessage = message.Copy();
963 upwardMessage.MetadataBlock.LevelFilter = MmLevelFilterHelper.SelfAndParents;
964 }
965 else if (needsChild)
966 {
967 downwardMessage = message.Copy();
968 downwardMessage.MetadataBlock.LevelFilter = MmLevelFilterHelper.SelfAndChildren;
969 }
970 // needsSelf uses original message unchanged
971 }
972 else
973 {
974 // No advanced routing - safe to reuse original message (lazy copy optimization)
975 if (needsParent)
976 {
977 message.MetadataBlock.LevelFilter = MmLevelFilterHelper.SelfAndParents;
978 upwardMessage = message;
979 }
980 else if (needsChild)
981 {
982 message.MetadataBlock.LevelFilter = MmLevelFilterHelper.SelfAndChildren;
983 downwardMessage = message;
984 }
985 // If needsSelf, message already has correct filter
986 }
987 }
988
989 // Second pass: invoke responders with appropriate messages
990 foreach (var routingTableItem in RoutingTable) {
991 var responder = routingTableItem.Responder;
992
993 // Skip responders that don't want handled messages (early termination optimization)
994 if (message.Handled && !responder.ReceiveHandledMessages)
995 continue;
996
997 MmLevelFilter responderLevel = routingTableItem.Level;
998
999 // FIX: Check if original levelFilter includes this responder's level
1000 // This prevents unwanted cascading when advanced routing sends with Self-only filter
1001 // Without this, lazy copy transforms Self SelfAndParents/SelfAndChildren and causes
1002 // cascading delivery to nodes that shouldn't receive the message
1003 if ((levelFilter & responderLevel) == 0)
1004 {
1005 continue; // Original levelFilter doesn't want to route to this responder's level
1006 }
1007
1008 // CRITICAL FIX: When advanced routing is active, skip Parent/Child responders
1009 // (Advanced routing via HandleAdvancedRouting will deliver to them)
1010 // This prevents double/multi-delivery bugs
1011 if (hasAdvancedFilters && (responderLevel & (MmLevelFilter.Parent | MmLevelFilter.Child)) != 0)
1012 {
1013 continue; // Skip - advanced routing will handle Parent/Child responders
1014 }
1015
1016 //Check individual responder level and then call the right param.
1017 MmMessage responderSpecificMessage;
1018 if((responderLevel & MmLevelFilter.Parent) > 0)
1019 {
1020 responderSpecificMessage = upwardMessage;
1021 }
1022 else if((responderLevel & MmLevelFilter.Child) > 0)
1023 {
1024 responderSpecificMessage = downwardMessage;
1025 }
1026 else
1027 {
1028 responderSpecificMessage = message;
1029 }
1030
1031 //MmLogger.LogFramework (gameObject.name + "observing " + responder.MmGameObject.name);
1032
1033 // CRITICAL: Use transformed filter from responderSpecificMessage, not captured levelFilter
1034 // (responderSpecificMessage may have transformed LevelFilter like SelfAndParents)
1035 bool checkPassed = ResponderCheck (responderSpecificMessage.MetadataBlock.LevelFilter,
1036 activeFilter, selectedFilter, networkFilter,
1037 routingTableItem, responderSpecificMessage);
1038
1039 if (checkPassed) {
1040 // Phase 5: Delegate dispatch optimization
1041 // Handler invocation: ~1-4 ticks, Virtual dispatch: ~8-10 ticks
1042 if (routingTableItem.Handler != null)
1043 routingTableItem.Handler(responderSpecificMessage);
1044 else
1045 responder.MmInvoke(responderSpecificMessage);
1046 }
1047 }
1048
1049 // Phase 2.1: Advanced Routing - Handle new level filters
1050 HandleAdvancedRouting(message, levelFilter);
1051
1052 //if (dirty && _prevMessageTime == message.TimeStamp)
1053 //{
1054 // dirty = false;
1055 //}
1056
1057 doNotModifyRoutingTable = false;
1058
1059 // Process queued responders (replaced Any() with Count for performance)
1060 while (MmRespondersToAdd.Count > 0)
1061 {
1062 var routingTableItem = MmRespondersToAdd.Dequeue();

1540 protected virtual bool ResponderCheck(MmLevelFilter levelFilter, MmActiveFilter activeFilter,
1541 MmSelectedFilter selectedFilter, MmNetworkFilter networkFilter,
1542 MmRoutingTableItem mmRoutingTableItem, MmMessage message)
1543 {
1544 // Q1-Q3: Use EnabledChecks flags to control which routing checks are performed.
1545 // When EnabledChecks == None, all messages pass to all responders (MmQuickNode behavior).
1546
1547 // Tag check (only if enabled and responders care about tags)
1548 if ((EnabledChecks & MmRoutingChecks.Tag) != 0
1549 && HasTagCheckEnabledResponders
1550 && message.MetadataBlock.Tag != MmTagHelper.Everything)
1551 {
1552 if (!TagCheck(mmRoutingTableItem, message)) return false;
1553 }
1554
1555 // Level check (only if enabled) - inline for common case
1556 if ((EnabledChecks & MmRoutingChecks.Level) != 0)
1557 {
1558 if ((levelFilter & mmRoutingTableItem.Level) == 0) return false;
1559 }
1560
1561 // Active check (only if enabled and filter is not All)
1562 if ((EnabledChecks & MmRoutingChecks.Active) != 0 && activeFilter != MmActiveFilter.All)
1563 {
1564 if (!ActiveCheck(activeFilter, mmRoutingTableItem.Responder)) return false;
1565 }
1566
1567 // Selected check (only if enabled) - MmRelaySwitchNode overrides this
1568 if ((EnabledChecks & MmRoutingChecks.Selected) != 0)
1569 {
1570 if (!SelectedCheck(selectedFilter, mmRoutingTableItem.Responder)) return false;
1571 }
1572
1573 // Network check (only if enabled and message is from network)
1574 if ((EnabledChecks & MmRoutingChecks.Network) != 0 && message.IsDeserialized)
1575 {
1576 if (!NetworkCheck(mmRoutingTableItem, message)) return false;
1577 }
1578
1579 return true;

1719 protected virtual void HandleAdvancedRouting(MmMessage message, MmLevelFilter levelFilter)
1720 {
1721 // Check if any advanced filters are present
1722 bool hasParent = (levelFilter & MmLevelFilter.Parent) != 0;
1723 bool hasSiblings = (levelFilter & MmLevelFilter.Siblings) != 0;
1724 bool hasCousins = (levelFilter & MmLevelFilter.Cousins) != 0;
1725 bool hasDescendants = (levelFilter & MmLevelFilter.Descendants) != 0;
1726 bool hasAncestors = (levelFilter & MmLevelFilter.Ancestors) != 0;
1727 bool hasCustom = (levelFilter & MmLevelFilter.Custom) != 0;
1728
1729 // If no advanced filters, skip (BEFORE profiling for zero overhead)
1730 if (!hasParent && !hasSiblings && !hasCousins && !hasDescendants && !hasAncestors && !hasCustom)
1731 return;
1732
1733 // Validate MmRoutingOptions if needed
1734 MmRoutingOptions options = message.MetadataBlock.Options;
1735
1736 // Validate lateral routing (siblings/cousins) requires AllowLateralRouting
1737 if ((hasSiblings || hasCousins) && (options == null || !options.AllowLateralRouting))
1738 {
1739 string errorMsg = $"[ROUTING] Lateral routing (Siblings/Cousins) requires MmRoutingOptions.AllowLateralRouting = true at '{name}'";
1740 if (StrictMode)
1741 {
1742 throw new System.InvalidOperationException(errorMsg);
1743 }
1744 MmLogger.LogFramework(errorMsg);
1745 return;
1746 }
1747
1748 // Validate custom filter requires CustomFilter predicate
1749 if (hasCustom && (options == null || options.CustomFilter == null))
1750 {
1751 string errorMsg = $"[ROUTING] Custom filter requires MmRoutingOptions.CustomFilter predicate at '{name}'";
1752 if (StrictMode)
1753 {
1754 throw new System.InvalidOperationException(errorMsg);
1755 }
1756 MmLogger.LogFramework(errorMsg);
1757 return;
1758 }
1759
1760 // Handle lateral routing (siblings/cousins)
1761 if (hasSiblings || hasCousins)
1762 {
1763 RouteLateral(message, hasSiblings, hasCousins);
1764 }
1765
1766 // Handle recursive descendants
1767 if (hasDescendants)
1768 {
1769 RouteRecursive(message, useDescendants: true);
1770 }
1771
1772 // Handle recursive ancestors
1773 if (hasAncestors)
1774 {
1775 RouteRecursive(message, useDescendants: false);
1776 }
1777
1778 // Handle direct parent routing
1779 if (hasParent)
1780 {
1781 HandleParentRouting(message);
1782 }
1783
1784 // Handle custom predicate filtering
1785 if (hasCustom && options != null && options.CustomFilter != null)
1786 {
