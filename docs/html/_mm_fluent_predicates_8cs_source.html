126 [MethodImpl(MethodImplOptions.AggressiveInlining)]
127 public bool Evaluate(MmRelayNode source, GameObject target)
128 {
129 if (target == null) return false;
130
131 switch (Type)
132 {
133 case MmPredicateType.Within:
134 if (source == null) return true;
135 float distance = Vector3.Distance(
136 source.transform.position,
137 target.transform.position
138 );
139 return distance <= Radius;
140
141 case MmPredicateType.InDirection:
142 if (source == null) return true;
143 Vector3 toTarget = (target.transform.position - source.transform.position).normalized;
144 float angleToTarget = Vector3.Angle(Direction, toTarget);
145 return angleToTarget <= Angle;
146
147 case MmPredicateType.InBounds:
148 return Bounds.Contains(target.transform.position);
149
150 case MmPredicateType.WithComponent:
151 return ComponentType != null && target.GetComponent(ComponentType) != null;
152
153 case MmPredicateType.Custom:
154 if (CustomFunc != null)
155 return CustomFunc(target);
156 if (CustomRelayFunc != null)
157 {
158 var relay = target.GetComponent<MmRelayNode>();
159 return relay != null && CustomRelayFunc(relay);
160 }
161 return true;
162
163 default:
164 return true;
165 }
166 }
