272 private static MmMethod DetectMethodFromPayload(object payload)
273 {
274 return payload switch
275 {
276 bool => MmMethod.MessageBool,
277 int => MmMethod.MessageInt,
278 float => MmMethod.MessageFloat,
279 string => MmMethod.MessageString,
280 Vector3 => MmMethod.MessageVector3,
281 Vector4 => MmMethod.MessageVector4,
282 Quaternion => MmMethod.MessageQuaternion,
283 Transform => MmMethod.MessageTransform,
284 GameObject => MmMethod.MessageGameObject,
285 MmMessage msg => msg.MmMethod,
286 null => MmMethod.NoOp,
287 _ => MmMethod.Message // Generic message for unknown types
288 };
289 }
