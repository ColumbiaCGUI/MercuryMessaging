60 internal bool TryInvoke(MmMessage message)
61 {
62 if (_disposed)
63 return false;
64
65 // Type check
66 if (!(message is T typedMessage))
67 return true; // Keep listening, just not our type
68
69 // Tag filter check
70 if (_tagFilter != MmTagHelper.Everything)
71 {
72 if ((message.MetadataBlock.Tag & _tagFilter) == 0)
73 return true; // Keep listening, tag doesn't match
74 }
75
76 // Custom filter check
77 if (_filter != null && !_filter(typedMessage))
78 return true; // Keep listening, filter didn't match
79
80 // Invoke the handler
81 try
82 {
83 _handler?.Invoke(typedMessage);
84 }
85 catch (Exception ex)
86 {
87 UnityEngine.Debug.LogError($"[MmListener] Exception in handler: {ex}");
88 }
89
90 // For one-time listeners, mark as disposed after handling
91 if (_oneTime)
92 {
93 _disposed = true;
94 return false; // Remove from registry
95 }
96
97 return true; // Keep listening
98 }

104 {
105 if (_disposed)
106 return;
107
108 _disposed = true;
109 _relay?.RemoveListener(this);
110 }
