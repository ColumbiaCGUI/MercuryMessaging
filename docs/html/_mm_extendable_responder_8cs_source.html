120 public override void MmInvoke(MmMessage message)
121 {
122 int methodValue = (int)message.MmMethod;
123
124 // FAST PATH: Standard framework methods (0-999)
125 // Route through MmBaseResponder's optimized switch statement
126 if (methodValue < 1000)
127 {
128 base.MmInvoke(message);
129 return;
130 }
131
132 // SLOW PATH: Custom methods (1000+)
133 // Route through registered handler dictionary
134 if (customHandlers != null &&
135 customHandlers.TryGetValue(message.MmMethod, out var handler))
136 {
137 try
138 {
139 handler(message);
140 }
141 catch (Exception ex)
142 {
143 Debug.LogError(
144 $"[{GetType().Name}] Error in custom handler for method {message.MmMethod} ({methodValue}): {ex}\n" +
145 $"GameObject: {gameObject.name}\n" +
146 $"Stack trace: {ex.StackTrace}");
147 }
148 }
149 else
150 {
151 OnUnhandledCustomMethod(message);
152 }
153 }

190 protected void RegisterCustomHandler(MmMethod method, Action<MmMessage> handler)
191 {
192 int methodValue = (int)method;
193
194 // Validate method ID is in custom range
195 if (methodValue < 1000)
196 {
197 throw new ArgumentException(
198 $"Custom methods must be >= 1000. Got: {method} ({methodValue}). " +
199 $"Values 0-999 are reserved for framework methods. " +
200 $"Please use a method ID of 1000 or higher.",
201 nameof(method));
202 }
203
204 // Validate handler is not null
205 if (handler == null)
206 {
207 throw new ArgumentNullException(nameof(handler),
208 $"Handler for method {method} ({methodValue}) cannot be null. " +
209 $"Please provide a valid Action<MmMessage> delegate.");
210 }
211
212 // Lazy-initialize dictionary on first registration
213 if (customHandlers == null)
214 {
215 customHandlers = new Dictionary<MmMethod, Action<MmMessage>>();
216 }
217
218 // Register handler (overwrites if already registered)
219 customHandlers[method] = handler;
220
221 MmLogger.LogFramework(
222 $"[{GetType().Name}] Registered custom handler for method {method} ({methodValue}) " +
223 $"on GameObject '{gameObject.name}'");
224 }
225
254 protected void UnregisterCustomHandler(MmMethod method)
255 {
256 if (customHandlers != null && customHandlers.Remove(method))
257 {
258 MmLogger.LogFramework(
259 $"[{GetType().Name}] Unregistered custom handler for method {method} ({(int)method}) " +
260 $"on GameObject '{gameObject.name}'");
261 }
262 }
263
290 protected bool HasCustomHandler(MmMethod method)
291 {
292 return customHandlers?.ContainsKey(method) == true;
293 }

335 protected virtual void OnUnhandledCustomMethod(MmMessage message)

340 $"GameObject: {gameObject.name}\n" +
341 $"Suggestion: Register a handler with RegisterCustomHandler() in Awake():\n" +
342 $" RegisterCustomHandler((MmMethod){methodValue}, OnYourHandlerMethod);");
343 }
