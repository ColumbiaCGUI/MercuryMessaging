219 public MmQuery<T> NamedLike(string pattern)
220 {
221 var regex = new System.Text.RegularExpressions.Regex(
222 "^" + System.Text.RegularExpressions.Regex.Escape(pattern).Replace("\\*", ".*") + "$",
223 System.Text.RegularExpressions.RegexOptions.IgnoreCase);
224
225 return Where(item =>
226 {
227 string name = null;
228 if (item is MmResponder responder)
229 name = responder.gameObject.name;
230 else if (item is Component component)
231 name = component.gameObject.name;
232 return name != null && regex.IsMatch(name);

344 private IEnumerable<T> EnumerateMatches()
345 {
346 if (_relay == null) yield break;
347
348 var visited = new HashSet<MmRelayNode>();
349
350 if (_recursive)
351 {
352 // Recursive traversal
353 foreach (var item in EnumerateRecursive(_relay, visited))
354 {
355 yield return item;
356 }
357 }
358 else
359 {
360 // Direct traversal of routing table
361 foreach (var item in EnumerateDirect(_relay))
362 {
363 yield return item;
364 }
365 }
366 }
367
371 private IEnumerable<T> EnumerateDirect(MmRelayNode relay)
372 {
373 if (relay?.RoutingTable == null) yield break;
374
375 foreach (var tableItem in relay.RoutingTable)
376 {
377 if (tableItem?.Responder == null) continue;
378
379 // Check level filter
380 if (!MatchesLevelFilter(tableItem.Level)) continue;
381
382 // Check active filter
383 if (_activeOnly && !tableItem.Responder.gameObject.activeInHierarchy) continue;
384 if (!_includeInactive && !tableItem.Responder.gameObject.activeInHierarchy) continue;
385
386 // Check tag filter
387 if (_tagFilter != MmTagHelper.Everything)
388 {
389 if (tableItem.Responder is MmBaseResponder baseResponder)
390 {
391 if ((baseResponder.Tag & _tagFilter) == 0) continue;
392 }
393 }
394
395 // Try to cast to target type
396 T typedItem = tableItem.Responder as T;
397 if (typedItem == null) continue;
398
399 // Check custom predicate
400 if (_predicate != null && !_predicate(typedItem)) continue;
401
402 yield return typedItem;
403 }
404 }
405
409 private IEnumerable<T> EnumerateRecursive(MmRelayNode relay, HashSet<MmRelayNode> visited)
410 {
411 if (relay == null || visited.Contains(relay)) yield break;
412 visited.Add(relay);
413
414 foreach (var item in EnumerateDirect(relay))
415 {
416 yield return item;
417 }
418
419 // Recurse into child relay nodes
420 if (relay.RoutingTable != null)
421 {
422 foreach (var tableItem in relay.RoutingTable)
423 {
424 if (tableItem?.Responder == null) continue;
425 if (!MatchesLevelFilter(tableItem.Level)) continue;
426
427 if (tableItem.Responder is MmRelayNode childRelay)
428 {
429 foreach (var childItem in EnumerateRecursive(childRelay, visited))
430 {
431 yield return childItem;
432 }
433 }
434 }
435 }
436 }

441 [MethodImpl(MethodImplOptions.AggressiveInlining)]
442 private bool MatchesLevelFilter(MmLevelFilter itemLevel)
443 {
444 // Handle special combined filters
445 if (_levelFilter == MmLevelFilterHelper.SelfAndChildren)
446 return itemLevel == MmLevelFilter.Self || itemLevel == MmLevelFilter.Child;
447
448 if (_levelFilter == MmLevelFilterHelper.SelfAndBidirectional)
449 return true; // Match all
450
451 // Direct comparison for simple filters
452 return (itemLevel & _levelFilter) != 0;
453 }
