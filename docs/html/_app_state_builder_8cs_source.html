62 public AppStateBuilder DefineState(string stateName)
63 {
64 if (string.IsNullOrEmpty(stateName))
65 throw new ArgumentException("State name cannot be empty", nameof(stateName));
66
67 _currentStateName = stateName;
68
69 if (!_stateConfigs.ContainsKey(stateName))

159 public MmRelaySwitchNode Build()
160 {
161 var fsm = _switchNode.RespondersFSM;
162 if (fsm == null)
163 {
164 Debug.LogWarning("[AppStateBuilder] RespondersFSM not initialized. Make sure switch node has been Awake'd.");
165 return _switchNode;
166 }
167
168 // Register state callbacks using the indexer
169 foreach (var kvp in _stateConfigs)
170 {
171 var stateName = kvp.Key;
172 var config = kvp.Value;
173
174 // Get the routing table item for this state
175 var routingItem = GetRoutingTableItem(stateName);
176 if (routingItem == null)
177 {
178 Debug.LogWarning($"[AppStateBuilder] State '{stateName}' not found in routing table");
179 continue;
180 }
181
182 // Get or create state events
183 var stateEvents = fsm[routingItem];
184 if (stateEvents == null)
185 {
186 stateEvents = new StateEvents();
187 fsm[routingItem] = stateEvents;
188 }
189
190 // Register enter callback
191 if (config.OnEnter != null)
192 {
193 var existingEnter = stateEvents.Enter;
194 stateEvents.Enter = () =>
195 {
196 existingEnter?.Invoke();
197 config.OnEnter();
198 };
199 }
200
201 // Register exit callback
202 if (config.OnExit != null)
203 {
204 var existingExit = stateEvents.Exit;
205 stateEvents.Exit = () =>
206 {
207 config.OnExit();
208 existingExit?.Invoke();
209 };
210 }
211 }
212
213 // Register global callbacks
214 if (_onGlobalEnter != null)
215 {
216 var existing = fsm.GlobalEnter;
217 fsm.GlobalEnter = () =>
218 {
219 existing?.Invoke();
220 _onGlobalEnter();
221 };
222 }
223
224 if (_onGlobalExit != null)
225 {
226 var existing = fsm.GlobalExit;
227 fsm.GlobalExit = () =>
228 {
229 _onGlobalExit();
230 existing?.Invoke();
231 };
232 }
233
234 // Set initial state
235 if (!string.IsNullOrEmpty(_initialState))
236 {
237 _switchNode.JumpTo(_initialState);
238 }
239
240 return _switchNode;
241 }
242
243 private MmRoutingTableItem GetRoutingTableItem(string stateName)
244 {
245 try
246 {
247 return _switchNode.RoutingTable[stateName];
248 }
249 catch (System.Collections.Generic.KeyNotFoundException)
250 {
251 // State not found in routing table - expected behavior for undefined states
252 Debug.LogWarning($"AppStateBuilder: State '{stateName}' not found in routing table");
253 return null;
254 }
255 catch (System.Exception e)
256 {
257 // Unexpected error - log full exception
258 Debug.LogError($"AppStateBuilder: Error getting routing table item for state '{stateName}'");
259 Debug.LogException(e);
260 return null;
261 }
262 }

264 #endregion
