50 {
51 private readonly Dictionary<uint, GameObject> _idToGameObject = new Dictionary<uint, GameObject>();
52 private readonly Dictionary<GameObject, uint> _gameObjectToId = new Dictionary<GameObject, uint>();
53 private uint _nextId = 1;
54
56 public string ResolverName => "RegistryResolver";

58 #region Registration
59
65 public uint Register(GameObject gameObject)
66 {
67 if (gameObject == null) return 0;
68
69 // Check if already registered
70 if (_gameObjectToId.TryGetValue(gameObject, out uint existingId))
71 {
72 return existingId;
73 }
74
75 uint id = _nextId++;
76 RegisterWithId(id, gameObject);
77 return id;
78 }
79
85 public void RegisterWithId(uint networkId, GameObject gameObject)
86 {
87 if (gameObject == null || networkId == 0) return;
88
89 // Remove any existing registrations
90 if (_idToGameObject.TryGetValue(networkId, out GameObject existing) && existing != gameObject)
91 {
92 _gameObjectToId.Remove(existing);
93 }
94 if (_gameObjectToId.TryGetValue(gameObject, out uint existingId) && existingId != networkId)
95 {
96 _idToGameObject.Remove(existingId);
97 }
98
99 _idToGameObject[networkId] = gameObject;
100 _gameObjectToId[gameObject] = networkId;
101
102 // Update next ID if necessary
103 if (networkId >= _nextId)
104 {
105 _nextId = networkId + 1;
106 }
107 }
108
113 public void Unregister(GameObject gameObject)
114 {
115 if (gameObject == null) return;
116
117 if (_gameObjectToId.TryGetValue(gameObject, out uint id))
118 {
119 _idToGameObject.Remove(id);
120 _gameObjectToId.Remove(gameObject);
121 }
122 }
123
128 public void Unregister(uint networkId)
129 {
130 if (_idToGameObject.TryGetValue(networkId, out GameObject gameObject))
131 {
132 _idToGameObject.Remove(networkId);
133 _gameObjectToId.Remove(gameObject);
134 }
135 }
136
140 public void Clear()
141 {
142 _idToGameObject.Clear();
143 _gameObjectToId.Clear();
144 _nextId = 1;
145 }
146
150 public int Count => _idToGameObject.Count;

157 public bool TryGetNetworkId(GameObject gameObject, out uint networkId)
158 {
159 if (gameObject == null)
160 {
161 networkId = 0;
162 return false;
163 }
164 return _gameObjectToId.TryGetValue(gameObject, out networkId);
165 }
166
168 public bool TryGetGameObject(uint networkId, out GameObject gameObject)
169 {
170 if (networkId == 0)
171 {
172 gameObject = null;
173 return false;
174 }
175
176 if (_idToGameObject.TryGetValue(networkId, out gameObject))
177 {
178 // Check if GameObject still exists
179 if (gameObject == null)
180 {
181 // Clean up stale reference
182 _idToGameObject.Remove(networkId);
183 return false;
184 }
185 return true;
186 }
187
188 gameObject = null;
189 return false;
190 }
191
193 public uint GetNetworkId(GameObject gameObject)
194 {
195 TryGetNetworkId(gameObject, out uint id);
196 return id;
197 }
198
200 public GameObject GetGameObject(uint networkId)
201 {
202 TryGetGameObject(networkId, out GameObject go);
203 return go;
204 }

207 public bool TryGetRelayNode(uint networkId, out MmRelayNode relayNode)
208 {

211 relayNode = go.GetComponent<MmRelayNode>();
212 return relayNode != null;

214
215 relayNode = null;
216 return false;
