108 public static void HandleKeyPress(KeyCode key, Modifier modifier)
109 {
110 Log(string.Format("Raw,Press,{0}", key));
111
112 KeyCode equivalentKey;
113 if (EquivalentKeyCode.TryGetValue(key, out equivalentKey)) {
114 HandleKeyPress(equivalentKey, modifier);
115 return;
116 }
117
118 Entry e;
119
120 // No entry
121 if (!Entries.TryGetValue(key, out e)) return;
122 // Not allowed
123 if (!e.IsAllowed) return;
124 // Shift modifier doesn't match
125 if (modifier != e.Modifier) return;
126
127 switch (e.ResponseTiming)
128 {
129 case RespondWhen.Released:
130 // This is an entry that should trigger on release, not press
131 return;
132 case RespondWhen.Clutched:
133 // This is a delayed keypress, initialize countdown
134 e.CountDown = Delay;
135 return;
136 default:
137 // All checks passed, respond
138 Log(string.Format("Cmd,Press,{0},{1}", key, e.Description));
139 Call(e.Response);
140 break;
141 }
142 }
143
144 public static void HandleKeyRelease(KeyCode key, Modifier modifier)
145 {
146 Log(string.Format("Raw,Release,{0}", key));
147
148 KeyCode equivalentKey;
149 if (EquivalentKeyCode.TryGetValue(key, out equivalentKey))
150 HandleKeyRelease(equivalentKey, modifier);
151
152 Entry e;
153
154 // No entry
155 if (!Entries.TryGetValue(key, out e)) return;
156 // Not allowed
157 if (!e.IsAllowed) return;
158 // Shift modifier doesn't match
159 if (modifier != e.Modifier) return;
160
161 switch (e.ResponseTiming)
162 {
163 case RespondWhen.Clutched:
164 // This is a delayed keypress
165 // CountDown isn't finished, we're aborting
166 if (e.CountDown > 0) {
167 // Reset countdown
168 e.CountDown = 0;
169 // Respond to abort
170 Log(string.Format("Cmd,Release,{0},{1},ClutchAbort", key, e.Description));
171 Call(e.ClutchAbortResponse);
172 }
173 else
174 {
175 // Countdown already finished, we can respond to release
176 if (e.ClutchReleaseResponse != null)
177 Log(string.Format("Cmd,Release,{0},{1},ClutchDone", key, e.Description));
178 Call(e.ClutchReleaseResponse);
179 }
180 break;
181 case RespondWhen.Pressed:
182 // This is an entry that should trigger on press, not release
183 return;
184 default:
185 // All checks passed, respond
186 Log(string.Format("Cmd,Release,{0},{1},Release", key, e.Description));
187 Call(e.Response);
188 break;
189 }
190 }

200 RespondWhen respondWhen = RespondWhen.Pressed,
201 Modifier modifier = Modifier.None,

203 UnityAction clutchReleaseResponse = null)
204 {
205 var e = new Entry

208 Description = description,
209 Response = response,
210 Modifier = modifier,
211 ResponseTiming = respondWhen,
212 ClutchAbortResponse = clutchAbortResponse,
213 ClutchReleaseResponse = clutchReleaseResponse,
214 IsAllowed = true
215 };
216 Entries[key] = e;
217 if (respondWhen == RespondWhen.Clutched) DelayedEntries.Add(e);
218 }

220 public static void ProcessDelayedEntries()
221 {
222 foreach (var de in DelayedEntries.Where(e => e.CountDown > 0))
223 {
224 de.CountDown--;
225 //Logger.Log.Debug(delayedEntry.CountDown);
226 if (de.CountDown != 0) continue;
227
228 Log(string.Format("Cmd,Delay,{0},{1}", de.Key, de.Description));
229
230 if (de.IsAllowed)
231 Call(de.Response);
232 }
233 }

244 public void Update() {
245
246 mods = Modifier.None;
247
248 if (UnityEngine.Input.GetKey(KeyCode.LeftAlt) || UnityEngine.Input.GetKey(KeyCode.RightAlt))
249 mods |= Modifier.Alt;
250
251 if (UnityEngine.Input.GetKey(KeyCode.LeftControl) || UnityEngine.Input.GetKey(KeyCode.RightControl))
252 mods |= Modifier.Ctrl;
253
254 if (UnityEngine.Input.GetKey(KeyCode.LeftShift) || UnityEngine.Input.GetKey(KeyCode.RightShift))
255 mods |= Modifier.Shift;
256
257 foreach (var e in Entries)
258 {
259 if(UnityEngine.Input.GetKeyDown(e.Key))
260 HandleKeyPress(e.Key, mods);
261 if(UnityEngine.Input.GetKeyUp(e.Key))
262 HandleKeyRelease(e.Key, mods);
263 }
264
265 foreach (var e in EquivalentKeyCode)
266 {
267 if(UnityEngine.Input.GetKeyDown(e.Value))
268 HandleKeyPress(e.Key, mods);
269 if(UnityEngine.Input.GetKeyUp(e.Value))
270 HandleKeyRelease(e.Key, mods);
271 }
272
273 ProcessDelayedEntries();
274
