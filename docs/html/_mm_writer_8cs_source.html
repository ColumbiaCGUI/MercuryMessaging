208 public void WriteString(string value)
209 {
210 if (string.IsNullOrEmpty(value))
211 {
212 WriteUShort(0);
213 return;
214 }
215
216 // Get required byte count
217 int byteCount = Utf8.GetByteCount(value);
218 if (byteCount > ushort.MaxValue)
219 {
220 throw new ArgumentException($"String too long: {byteCount} bytes (max {ushort.MaxValue})");
221 }
222

224 EnsureCapacity(byteCount);

226 // Encode directly into buffer

347 #region Capacity Management
348
349 private void EnsureCapacity(int additionalBytes)
350 {
351 int required = _position + additionalBytes;
352 if (required <= _buffer.Length) return;
353
354 // Calculate new size (double until sufficient)
355 int newSize = _buffer.Length;
356 while (newSize < required && newSize < MaxCapacity)
357 {
358 newSize *= 2;
359 }
360
361 if (newSize > MaxCapacity)
362 {
363 newSize = MaxCapacity;
364 }
365
366 if (required > newSize)
367 {
368 throw new InvalidOperationException($"MmWriter buffer overflow: required {required} bytes, max {MaxCapacity}");
369 }
370
371 // Rent new buffer and copy data
372 byte[] newBuffer = ArrayPool<byte>.Shared.Rent(newSize);
373 Buffer.BlockCopy(_buffer, 0, newBuffer, 0, _position);
374
375 // Return old buffer to pool
376 ArrayPool<byte>.Shared.Return(_buffer);
377 _buffer = newBuffer;
378 }
379
380 #endregion

384 public void Dispose()
385 {

388

395 }

403 internal class ObjectPool<T> where T : class
404 {
405 private readonly Func<T> _factory;
406 private readonly Action<T> _reset;
407 private readonly Stack<T> _pool;
408 private readonly object _lock = new object();
409 private const int MaxPoolSize = 16;
410
411 public ObjectPool(Func<T> factory, Action<T> reset)
412 {
413 _factory = factory;
414 _reset = reset;
415 _pool = new Stack<T>();
416 }
417
418 public T Get()
419 {
420 lock (_lock)
421 {
422 if (_pool.Count > 0)
423 {
424 T item = _pool.Pop();
425 _reset(item);
426 return item;
427 }
428 }
429 return _factory();
430 }
431
432 public void Return(T item)
433 {
434 lock (_lock)
435 {
436 if (_pool.Count < MaxPoolSize)
437 {
438 _pool.Push(item);
439 }
440 }
441 }
442 }
