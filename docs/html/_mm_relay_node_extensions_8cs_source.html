376 private static readonly Dictionary<int, Action<MmMessage>> _pendingQueries =
377 new Dictionary<int, Action<MmMessage>>();
378 private static int _nextQueryId = 1;
379
383 public static int Query(this MmRelayNode relay, MmMethod method, Action<MmMessage> onResponse)
384 {
385 int queryId = _nextQueryId++;
386 _pendingQueries[queryId] = onResponse;
387
388 var metadata = new MmMetadataBlock(
389 MmLevelFilterHelper.SelfAndChildren,
390 MmActiveFilter.All,
391 MmSelectedFilter.All,
392 MmNetworkFilter.Local
393 );
394 var queryMessage = MmMessagePool.GetInt(queryId, method, metadata);
395
396 relay.MmInvoke(queryMessage);
397 MmMessagePool.Return(queryMessage);
398 return queryId;
399 }
400
404 public static int QueryDescendants(this MmRelayNode relay, MmMethod method, Action<MmMessage> onResponse)
405 {
406 int queryId = _nextQueryId++;
407 _pendingQueries[queryId] = onResponse;
408
409 // Use fluent API for proper descendant routing
410 new MmFluentMessage(relay, method, queryId).ToDescendants().Execute();
411 return queryId;
412 }

417 public static void Respond(this MmRelayNode relay, int queryId, MmMessage response)
418 {

420 {

481 private static MmRelayNode FindTargetRelay(MmRelayNode source, string targetName)
482 {
483 if (source == null || string.IsNullOrEmpty(targetName))
484 return null;
485
486 // Check self first
487 if (source.gameObject.name == targetName)
488 return source;
489
490 // Search in routing table (children)
491 if (source.RoutingTable != null)
492 {
493 foreach (var item in source.RoutingTable)
494 {
495 if (item.Level != MmLevelFilter.Child)
496 continue;
497
498 var childRelay = item.Responder?.GetRelayNode();
499 if (childRelay != null)
500 {
501 if (childRelay.gameObject.name == targetName)
502 return childRelay;
503
504 // Recursive search
505 var found = FindTargetRelay(childRelay, targetName);
506 if (found != null)
507 return found;
508 }
509 }
510 }
511
512 return null;
513 }
