127 public override object[] Serialize()

129 object[] baseSerialized = base.Serialize();

131 // Pre-allocate combined array: base + 1 (length) + byte array length
132 // This fixes O(nÂ²) complexity from the foreach + Concat pattern
133 object[] result = new object[baseSerialized.Length + 1 + byteArr.Length];
134
135 // Copy base data using Array.Copy (no LINQ)
136 Array.Copy(baseSerialized, 0, result, 0, baseSerialized.Length);
137
138 // Fill length
139 result[baseSerialized.Length] = byteArr.Length;
140
141 // Fill byte array directly (no loop concatenation)
