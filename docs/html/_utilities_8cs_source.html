72 public static double PositiveAngleBetween(Vector2 v1, Vector2 v2)
73 {
74 var angle = AngleBetween(v1, v2);
75 // Atan2 returns a 0 to (-180) angle for clockwise rotations and 0 to 180 angle for counter-clockwise
76 // We convert it to 0-360 clockwise
77 if (angle < 0) return angle * -1;
78 return (2*Mathf.PI) - angle;
79 }

81 public static double AngleBetween(Vector2 v1, Vector2 v2)
82 {
83 // angle = mod(atan2(x1*y2-x2*y1,x1*x2+y1*y2),2*pi);
84 return System.Math.Atan2(v1.x * v2.y - v2.x * v1.y, v1.x * v2.x + v1.y * v2.y);

87 public static Vector3 QuaternionToYawPitchRoll(Quaternion q)
88 {
89
90 const float epsilon = 0.0009765625f;
91 const float threshold = 0.5f - epsilon;
92
93 float yaw;
94 float pitch;
95 float roll;
96
97 var xy = q.x * q.y;
98 var zw = q.z * q.w;
99
100 var test = xy + zw;
101
102 if (test < -threshold || test > threshold)
103 {
104
105 var sign = System.Math.Sign(test);
106
107 yaw = sign * 2 * (float)System.Math.Atan2(q.x, q.w);
108
109 pitch = sign * Mathf.PI/2;
110
111 roll = 0;
112
113 }
114 else
115 {
116
117 var xx = q.x * q.x;
118 var xz = q.x * q.z;
119 var xw = q.x * q.w;
120
121 var yy = q.y * q.y;
122 var yw = q.y * q.w;
123 var yz = q.y * q.z;
124
125 var zz = q.z * q.z;
126
127 yaw = (float)System.Math.Atan2(2 * yw - 2 * xz, 1 - 2 * yy - 2 * zz);
128
129 pitch = (float)System.Math.Atan2(2 * xw - 2 * yz, 1 - 2 * xx - 2 * zz);
130
131 roll = (float)System.Math.Asin(2 * test);
132
133 }
134
135 return new Vector3(yaw, pitch, roll);
136 }

138 public static EulerAnglesSimple QuaternionToYawPitchRoll2(Quaternion q)
139 {
140 double ww = q.w * q.w;
141 double xx = q.x * q.x;
142 double yy = q.y * q.y;
143 double zz = q.z * q.z;
144
145 double yz = q.y * q.z;
146 double xz = q.x * q.z;
147 double xw = q.x * q.w;
148 double yw = q.y * q.w;
149 double xy = q.x * q.y;
150 double zw = q.z * q.w;
151
152 var rotxrad = System.Math.Atan2(2.0 * (yz + xw), (-xx - yy + zz + ww));
153 var rotyrad = System.Math.Asin(-2.0 * (xz - yw));
154 var rotzrad = System.Math.Atan2(2.0 * (xy + zw), (xx - yy - zz + ww));
155
156 return new EulerAnglesSimple(rotxrad, rotyrad, rotzrad);
157 }

212 public static T[] Shuffle<T>(T[] iArray)
213 {
214 T[] array = new T[iArray.Length];
215 iArray.CopyTo(array, 0);
216
217 System.Random Rnd = new System.Random(System.DateTime.Now.Millisecond);
218
219 var n = array.Length;
220 while (n > 1)
221 {
222 var k = Rnd.Next(n--);
223 var temp = array[n];
224 array[n] = array[k];
225 array[k] = temp;
226 }
227
228 return array;
229 }
