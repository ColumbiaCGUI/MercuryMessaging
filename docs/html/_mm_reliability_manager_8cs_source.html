164 public byte[] WrapPacket(byte[] data, MmReliability reliability, int targetClient)
165 {
166 uint seq = GetNextSequence();
167
168 var header = new MmPacketHeader
169 {
170 PacketType = MmPacketType.Data,
171 SequenceNumber = seq,
172 AckBitfield = _receivedBitfield,
173 AckSequence = _lastReceivedSequence,
174 Reliability = reliability
175 };
176
177 using (var ms = new MemoryStream())
178 using (var writer = new BinaryWriter(ms))
179 {
180 header.Write(writer);
181 writer.Write(data);
182 var wrappedData = ms.ToArray();
183
184 // Track reliable packets for retransmission
185 if (reliability == MmReliability.Reliable)
186 {
187 lock (_lock)
188 {
189 _pendingPackets[seq] = new MmPendingPacket
190 {
191 SequenceNumber = seq,
192 Data = wrappedData,
193 SendTime = Time.timeAsDouble,
194 RetryCount = 0,
195 Reliability = reliability,
196 TargetClient = targetClient

208 public byte[] UnwrapPacket(byte[] packet, out MmPacketHeader header)
209 {

212 {

301 public void ProcessPong(byte[] packet)
302 {
303 using (var ms = new MemoryStream(packet))
304 using (var reader = new BinaryReader(ms))
305 {
306 var header = MmPacketHeader.Read(reader);
307 if (header.PacketType != MmPacketType.Pong) return;
308
309 double sendTime = reader.ReadDouble();
310 float rtt = (float)(Time.timeAsDouble - sendTime);
311 UpdateRTT(rtt);

318 public void UpdateRTT(float measuredRTT)
319 {
320 if (SmoothedRTT <= 0)
321 {
322 SmoothedRTT = measuredRTT;
323 RTTVariance = measuredRTT / 2;
324 }
325 else
326 {
327 float rttDiff = Mathf.Abs(SmoothedRTT - measuredRTT);
328 RTTVariance = (1 - RTTAlpha) * RTTVariance + RTTAlpha * rttDiff;
329 SmoothedRTT = (1 - RTTAlpha) * SmoothedRTT + RTTAlpha * measuredRTT;
330 }
331 }

337 public void Update(double currentTime)
338 {
339 List<MmPendingPacket> toRetransmit = null;
340 List<uint> toLost = null;
341
342 lock (_lock)
343 {
344 foreach (var kvp in _pendingPackets)
345 {
346 var pending = kvp.Value;
347 float timeout = CurrentRTO * (1 << pending.RetryCount); // Exponential backoff
348
349 if (currentTime - pending.SendTime > timeout)
350 {
351 if (pending.RetryCount >= MaxRetries)
352 {
353 // Give up on this packet
354 if (toLost == null) toLost = new List<uint>();
355 toLost.Add(kvp.Key);
356 }
357 else
358 {
359 // Queue for retransmission
360 if (toRetransmit == null) toRetransmit = new List<MmPendingPacket>();
361 toRetransmit.Add(pending);
362 }
363 }
364 }
365
366 // Remove lost packets
367 if (toLost != null)
368 {
369 foreach (var seq in toLost)
370 {
371 _pendingPackets.Remove(seq);
372 }
373 }
374 }
375
376 // Fire events outside lock
377 if (toLost != null)
378 {
379 foreach (var seq in toLost)
380 {
381 OnPacketLost?.Invoke(seq);
382 }
383 }
384
385 if (toRetransmit != null)
386 {
387 foreach (var pending in toRetransmit)
388 {
389 pending.RetryCount++;
390 pending.SendTime = currentTime;
391 OnRetransmit?.Invoke(pending.Data, pending.TargetClient);
392 }
393 }
394 }

407 private void TrackReceived(uint seq)
408 {
409 if (seq > _lastReceivedSequence)
410 {
411 // Shift bitfield to accommodate new sequence
412 uint shift = seq - _lastReceivedSequence;
413 if (shift < 32)
414 {
415 _receivedBitfield = (_receivedBitfield << (int)shift) | 1;
416 }
417 else
418 {
419 _receivedBitfield = 1;
420 }
421 _lastReceivedSequence = seq;
422 }
423 else if (_lastReceivedSequence - seq < 32)
424 {
425 // Mark older sequence as received
426 uint bit = _lastReceivedSequence - seq;
427 _receivedBitfield |= (1u << (int)bit);
428 }
429 }

434 private void ProcessAcks(uint ackSeq, uint ackBits)
435 {
436 List<uint> acked = null;
437
438 lock (_lock)
439 {
440 // Check main ack sequence
441 if (_pendingPackets.ContainsKey(ackSeq))
442 {
443 if (acked == null) acked = new List<uint>();
444 acked.Add(ackSeq);
445 _pendingPackets.Remove(ackSeq);
446 }
447
448 // Check bitfield for additional acks
449 for (int i = 0; i < 32; i++)
450 {
451 if ((ackBits & (1u << i)) != 0)
452 {
453 uint seq = ackSeq - (uint)i - 1;
454 if (_pendingPackets.ContainsKey(seq))
455 {
456 if (acked == null) acked = new List<uint>();
457 acked.Add(seq);
458 _pendingPackets.Remove(seq);
459 }
460 }
461 }
462 }
463
464 // Fire events outside lock
465 if (acked != null)
466 {
467 foreach (var seq in acked)
468 {
469 OnPacketAcked?.Invoke(seq);
470 }
471 }
472 }
