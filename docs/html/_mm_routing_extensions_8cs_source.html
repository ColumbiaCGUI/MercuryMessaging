237 private static void RemoveFromRoutingTable(MmRelayNode parent, MmResponder responder)
238 {
239 if (parent?.RoutingTable == null || responder == null) return;
240
241 // Find and remove the item from routing table
242 for (int i = parent.RoutingTable.Count - 1; i >= 0; i--)
243 {
244 if (parent.RoutingTable[i]?.Responder == responder)
245 {
246 parent.RoutingTable.RemoveAt(i);
247 break; // Only remove first match
248 }
249 }
250 }
