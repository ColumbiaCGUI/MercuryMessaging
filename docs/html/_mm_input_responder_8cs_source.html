65 public override void MmInvoke(MmMessage message)
66 {
67 int methodValue = (int)message.MmMethod;
68
69 // Input messages are in range 200-299
70 if (methodValue >= 200 && methodValue < 300)
71 {
72 DispatchInputMessage(message);
73 return;
74 }
75
76 // All other messages go to base class
77 base.MmInvoke(message);
78 }

83 private void DispatchInputMessage(MmMessage message)
84 {
85 switch ((MmInputMethod)(int)message.MmMethod)
86 {
87 case MmInputMethod.Input6DOF:
88 Received6DOF((MmInput6DOFMessage)message);
89 break;
90
91 case MmInputMethod.Gesture:
92 ReceivedGesture((MmInputGestureMessage)message);
93 break;
94
95 case MmInputMethod.Haptic:
96 ReceivedHaptic((MmInputHapticMessage)message);
97 break;
98
99 case MmInputMethod.Button:
100 ReceivedButton((MmInputButtonMessage)message);
101 break;
102
103 case MmInputMethod.Axis:
104 ReceivedAxis((MmInputAxisMessage)message);
105 break;
106
107 case MmInputMethod.Touch:
108 ReceivedTouch((MmInputTouchMessage)message);
109 break;
110
111 case MmInputMethod.ControllerState:
112 ReceivedControllerState((MmInputControllerStateMessage)message);
113 break;
114
115 case MmInputMethod.Gaze:
116 ReceivedGaze((MmInputGazeMessage)message);
117 break;
118
119 default:
120 OnUnhandledInputMethod(message);
121 break;
122 }
