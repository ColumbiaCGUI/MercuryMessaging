39namespace MercuryMessaging.StandardLibrary.Animation
40{
41 public class MaterialChanger<T, TI> : MonoBehaviour
42 where TI : Interpolator<T>, InterpolatorMaterial<T>
43 {
44 public T Target;
45 public float Duration = 1;
46
47 readonly Dictionary<T, TI> interpolators = new Dictionary<T, TI>();
48
49 TI selectedInterpolator;
50
51 enum MaterialState
52 {
53 Normal,
54 Changed
55 }
56
57 readonly FiniteStateMachine<MaterialState> materialFSM
58 = new FiniteStateMachine<MaterialState>("MaterialState");
59
60 MeshRenderer[] meshRenderers;
61
62 public virtual void Awake()
63 {
64 meshRenderers = GetComponentsInChildren<MeshRenderer>();
65
66 materialFSM[MaterialState.Normal].Exit = delegate
67 {
68 foreach (var interpolator in interpolators.Values)
69 {
70 interpolator.Forward();
71 interpolator.Begin();
72 }
73 };
74
75 materialFSM[MaterialState.Changed].Exit = delegate
76 {
77 foreach (var interpolator in interpolators.Values)
78 {
79 interpolator.Backward();
80 interpolator.Begin();
81 }
82 };
83
84 KeyboardHandler.AddEntry(KeyCode.J, "Start Material Changer", Change);
85 KeyboardHandler.AddEntry(KeyCode.K, "Reverse Material Changer", Reverse);
86 KeyboardHandler.AddEntry(KeyCode.H, "Cleanup Material Changer", Cleanup);
87
88 Refresh();
89 }
90
91 public virtual void Change()
92 {
93 materialFSM.JumpTo(MaterialState.Changed);
94 }
95
96 public virtual void Reverse()
97 {
98 materialFSM.JumpTo(MaterialState.Normal);
99 }
100
101 public virtual void Refresh()
102 {
103 foreach (var interpolator in interpolators.Values)
104 {
105 interpolator.Materials.Clear();
106 }
107
108 interpolators.Clear();
109
110 foreach (var meshRenderer in meshRenderers)
111 {
112 foreach (var material in meshRenderer.materials)
113 {
114 var first = (selectedInterpolator == null);
115 var addNew = false;
116
117 if (first)
118 {
119 selectedInterpolator = gameObject.AddComponent<TI>();
120 }
121
122 var original = selectedInterpolator.GetValue(material);
123
124 //Debug.LogFormat("original: {0}", original);
125
126 // If it is not the first, check if it exists in dictionary
127 if (!first)
128 {
129 addNew = !interpolators.TryGetValue(original, out selectedInterpolator);
130
131 // If it is doesn't exist in dictionary, create a new component
132 if (addNew)
133 selectedInterpolator = gameObject.AddComponent<TI>();
134 }
135
136 // If it is the first one or if we just created a new one and added it
137 // to the dictionary, set it up
138 if (first || addNew)
139 {
140 selectedInterpolator.ValueOriginal = original;
141 selectedInterpolator.ValueTarget = Target;
142 selectedInterpolator.Duration = Duration;
143 selectedInterpolator.Materials = new List<Material>();
144 interpolators[original] = selectedInterpolator;
145 }
146
147 selectedInterpolator.Materials.Add(material);
148 }
149 }
150 }
151
152 public virtual void Cleanup()
153 {
154 foreach (var interpolator in interpolators.Values)
155 {
156 interpolator.Revert();
157 interpolator.Materials.Clear();
158 Destroy(interpolator);
159 }
160 }
161 }
162}
