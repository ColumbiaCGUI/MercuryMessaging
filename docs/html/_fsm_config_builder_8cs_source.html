116 public FsmConfigBuilder<T> OnEnter(string stateName, Action callback)
117 {
118 if (_stateResolver == null)
119 throw new InvalidOperationException("State resolver not set. Use OnEnter(T state, Action) instead.");
120
121 _deferredActions.Add(() =>
122 {
123 var state = _stateResolver(stateName);
124 if (state != null && _fsm[state] != null)
125 _fsm[state].Enter = callback != null ? new Transition(callback) : null;
126 });
127 return this;
128 }

174 public FsmConfigBuilder<T> StartWith(string stateName)
175 {
176 if (_stateResolver == null)
177 throw new InvalidOperationException("State resolver not set. Use StartWith(T state) instead.");
178
179 _deferredActions.Add(() =>

181 var state = _stateResolver(stateName);

190 #region Build
191
195 public void Build()
196 {
197 // Apply global handlers
198 if (_globalEnter != null)
199 _fsm.GlobalEnter = _globalEnter;
200 if (_globalExit != null)
201 _fsm.GlobalExit = _globalExit;
202
203 // Execute all deferred actions (state-specific handlers)
204 foreach (var action in _deferredActions)
205 {
206 try
207 {
208 action();
209 }
210 catch (Exception ex)
211 {
212 UnityEngine.Debug.LogWarning($"[FsmConfigBuilder] Error applying configuration: {ex.Message}");
213 }
214 }
215 }
