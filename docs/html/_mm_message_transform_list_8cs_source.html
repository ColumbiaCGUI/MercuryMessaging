107 public override int Deserialize(object[] data)
108 {
109 int index = base.Deserialize(data);
110 int numTransforms = (int) data[index++];
111 transforms = new List<MmTransform>();
112 for (int i = 0; i < numTransforms; i++)
113 {
114 MmTransform transform = new MmTransform();
115 index = transform.Deserialize(data, index);
116 transforms.Add(transform);
117 }
118 return index;
119 }

130 public override object[] Serialize()
131 {
132 object[] baseSerialized = base.Serialize();
133
134 // MmTransform serializes to 6 elements: Translation(1) + Rotation(4) + Scale(1)
135 const int TRANSFORM_SIZE = 6;
136
137 // Pre-allocate combined array: base + 1 (count) + transforms * TRANSFORM_SIZE
138 // This fixes O(nÂ²) complexity from the foreach + Concat pattern
139 object[] result = new object[baseSerialized.Length + 1 + transforms.Count * TRANSFORM_SIZE];
140
141 // Copy base data using Array.Copy (no LINQ)
142 Array.Copy(baseSerialized, 0, result, 0, baseSerialized.Length);
143
144 // Fill count
145 result[baseSerialized.Length] = transforms.Count;
146
147 // Fill transforms directly (no loop concatenation)
148 int idx = baseSerialized.Length + 1;
149 foreach (MmTransform transform in transforms)
150 {
151 object[] transformSerialized = transform.Serialize();
152 Array.Copy(transformSerialized, 0, result, idx, transformSerialized.Length);
153 idx += transformSerialized.Length;
154 }
155
156 return result;
157 }
