227 MmMessageType = (MercuryMessaging.MmMessageType) (short)data[index++];

246 public virtual object[] Serialize()
247 {
248 object[] metadataSerialized = MetadataBlock.Serialize();
249
250 // Pre-allocate combined array: 3 base + metadata length
251 object[] result = new object[3 + metadataSerialized.Length];
252
253 // Fill base data directly
254 result[0] = (short)MmMethod;
255 result[1] = (short)MmMessageType;
256 result[2] = (int)NetId;
257
258 // Copy metadata using Array.Copy (no LINQ)
259 Array.Copy(metadataSerialized, 0, result, 3, metadataSerialized.Length);
260
261 return result;
262 }
