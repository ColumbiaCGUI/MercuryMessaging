100 private List<MmRoutingTableItem> GetChildRelayNodes()
101 {
102 _fsmStateBuffer.Clear();

105 if (item.Responder is MmRelayNode && item.Level == MmLevelFilter.Child)

107 _fsmStateBuffer.Add(item);
