81 public static T[] GetComponentsInChildrenWithTag<T>(this GameObject gameObject, string tag)
82 where T : Component
83 {
84 List<T> results = new List<T>();
85
86 if (gameObject.CompareTag(tag))
87 results.Add(gameObject.GetComponent<T>());
88
89 foreach (Transform t in gameObject.transform)
90 results.AddRange(t.gameObject.GetComponentsInChildrenWithTag<T>(tag));
91
92 return results.ToArray();
93 }
