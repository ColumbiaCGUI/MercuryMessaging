192 private void ResolveTargetNetId()
193 {
194#if FUSION2_AVAILABLE
195 var bridge = MmNetworkBridge.Instance;
196
197 // Debug: Log bridge state
198 AddLog($"ResolveTargetNetId: Bridge={bridge != null}, Resolver={bridge?.Resolver != null}");
199
200 // PRIORITY: Check if TestRoot exists (from Build Test Hierarchy)
201 var testRoot = GameObject.Find("TestRoot");
202 if (testRoot != null)
203 {
204 var testRootRelay = testRoot.GetComponent<MmRelayNode>();
205 if (testRootRelay != null)
206 {
207 targetRelayNode = testRootRelay;
208 targetNetId = GetDeterministicId(testRoot);
209 AddLog($"Target set to TestRoot (NetId: {targetNetId}) - use 'Build Test Hierarchy' results");
210 return;
211 }
212 }
213
214 // Try to get NetId from targetRelayNode (Inspector-assigned)
215 if (targetRelayNode != null && targetRelayNode.gameObject.activeInHierarchy)
216 {
217 if (bridge?.Resolver != null && bridge.Resolver.TryGetNetworkId(targetRelayNode.gameObject, out uint netId))
218 {
219 targetNetId = netId;
220 AddLog($"Target NetId resolved: {targetNetId} ({targetRelayNode.gameObject.name})");
221 return;
222 }
223
224 // Fallback: Use deterministic path-based hash (matches Fusion2BridgeSetup)
225 targetNetId = GetDeterministicId(targetRelayNode.gameObject);
226 AddLog($"Target NetId (path-based): {targetNetId} ({targetRelayNode.gameObject.name})");
227 return;
228 }
229
230 // Try to find any MmRelayNode in scene, INCLUDING inactive ones
231 var relayNodes = FindObjectsByType<MmRelayNode>(FindObjectsInactive.Include, FindObjectsSortMode.None);
232 AddLog($"Found {relayNodes.Length} MmRelayNode(s) in scene (including inactive)");
233
234 // Try to get NetworkObject ID first (for spawned objects with valid Id > 0)
235 foreach (var node in relayNodes)
236 {
237 if (bridge?.Resolver != null && bridge.Resolver.TryGetNetworkId(node.gameObject, out uint netId))
238 {
239 targetNetId = netId;
240 targetRelayNode = node;
241 AddLog($"Auto-resolved target NetId: {targetNetId} ({node.gameObject.name})");
242 return;
243 }
244 }
245
246 // Fallback: Use first relay node with deterministic ID (path-based)
247 if (relayNodes.Length > 0)
248 {
249 var firstNode = relayNodes[0];
250 targetNetId = GetDeterministicId(firstNode.gameObject);
251 targetRelayNode = firstNode;
252 AddLog($"Auto-resolved target (path-based): {targetNetId} ({firstNode.gameObject.name})");
253 return;
254 }
255
256 AddLog("No relay nodes found - messages will not route to responders");
257#endif
258 }

263 private void EnsureNetworkTestResponder()
264 {
265 if (targetRelayNode == null) return;
266
267 // Check if NetworkTestResponder already exists
268 var responder = targetRelayNode.GetComponent<NetworkTestResponder>();
269 if (responder != null)
270 {
271 AddLog($"[OK] {targetRelayNode.name} has NetworkTestResponder");
272 return;
273 }
274
275 // Remove plain MmBaseResponder if present (it doesn't log messages)
276 var baseResponder = targetRelayNode.GetComponent<MmBaseResponder>();
277 if (baseResponder != null && baseResponder.GetType() == typeof(MmBaseResponder))
278 {
279 Destroy(baseResponder);
280 AddLog($"Removed plain MmBaseResponder from {targetRelayNode.name}");
281 }
282
283 // Add NetworkTestResponder
284 targetRelayNode.gameObject.AddComponent<NetworkTestResponder>();
285 AddLog($"Added NetworkTestResponder to {targetRelayNode.name}");
286
287 // Refresh routing table after adding component
288 StartCoroutine(RefreshAfterFrame());
289 }

329 private uint GetDeterministicId(GameObject go)
330 {
331 string path = GetHierarchyPath(go);
332 int hash = path.GetHashCode();
333 return (uint)((hash & 0x7FFFFFFF) | 1);
334 }
335
336 private string GetHierarchyPath(GameObject go)
337 {
338 var path = go.name;
339 var parent = go.transform.parent;
340 while (parent != null)
341 {
342 path = parent.name + "/" + path;
343 parent = parent.parent;
344 }
345 return path;
346 }

380 private void OnBridgeMessageReceived(MmMessage message)
381 {
382 messagesReceived++;
383 string info = $"Bridge received {message.MmMessageType} (NetId: {message.NetId})";
384
385 // Extract value based on type
386 switch (message.MmMessageType)
387 {
388 case MmMessageType.MmString:
389 info += $": \"{((MmMessageString)message).value}\"";
390 break;
391 case MmMessageType.MmInt:
392 info += $": {((MmMessageInt)message).value}";
393 break;
394 case MmMessageType.MmFloat:
395 info += $": {((MmMessageFloat)message).value}";
396 break;
397 }
398
399 // Check if it will route to a responder - verify actual responder presence
400 if (message.NetId != 0 && MmNetworkBridge.Instance.TryGetRelayNode((uint)message.NetId, out var node))
401 {
402 var responderCount = node.GetComponentsInChildren<NetworkTestResponder>().Length;
403 if (responderCount > 0)
404 {
405 messagesRoutedToResponders++;
406 info += $" -> Routed to {responderCount} responder(s)";
407 }
408 else
409 {
410 info += " -> Node found but NO responders attached!";
411 }
412 }
413 else if (message.NetId != 0)
414 {
415 info += " -> No relay node for NetId";
416 }
417
418 AddLog(info);
419 }
