189 private static IEnumerator RepeatingInvokeCoroutine(MmRelayNode relay, float interval, Action action, int repeatCount, CancellationToken token)
190 {
191 var wait = new WaitForSeconds(interval);
192 int count = 0;
193
194 while (!token.IsCancellationRequested && relay != null)
195 {
196 yield return wait;
197
198 if (token.IsCancellationRequested || relay == null)
199 break;
200
201 action?.Invoke();
202
203 if (repeatCount > 0)
204 {
205 count++;
206 if (count >= repeatCount)
207 break;
208 }
209 }
210 }

261 private static IEnumerator ConditionalInvokeCoroutine(MmRelayNode relay, Func<bool> condition, Action action, float timeout, CancellationToken token)
262 {
263 float elapsed = 0f;
264
265 while (!token.IsCancellationRequested && relay != null)
266 {
267 if (condition != null && condition())
268 {
269 action?.Invoke();
270 yield break;
271 }
272
273 yield return null;
274 elapsed += Time.deltaTime;
275
276 if (timeout > 0 && elapsed >= timeout)
277 {
278 Debug.LogWarning("MmTemporalExtensions.When: Condition timed out");
279 yield break;
280 }
281 }

314 public static async Task<T> RequestAsync<T>(
315 this MmRelayNode relay,
316 MmMethod method,
317 MmMetadataBlock metadata,
318 float timeout = 5f,
319 CancellationToken cancellationToken = default) where T : MmMessage
320 {
321 var tcs = new TaskCompletionSource<T>();
322 int queryId = RegisterAsyncQuery<T>(tcs);
323
324 // Set up timeout
325 var cts = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken);
326 cts.CancelAfter(TimeSpan.FromSeconds(timeout));
327
328 // Register cancellation
329 cts.Token.Register(() =>
330 {
331 UnregisterAsyncQuery(queryId);
332 tcs.TrySetCanceled();
333 });
334
335 // Send the query (using pool for efficiency)
336 var queryMessage = MmMessagePool.GetInt(queryId, method, metadata);
337 relay.MmInvoke(queryMessage);
338 MmMessagePool.Return(queryMessage);
339
340 try
341 {
342 return await tcs.Task;
343 }
344 finally
345 {
346 cts.Dispose();
347 }
348 }

359 private static readonly Dictionary<int, object> _asyncQueries = new Dictionary<int, object>();
360 private static int _asyncQueryId = 1;
361 private static readonly object _queryLock = new object();
362
363 private static int RegisterAsyncQuery<T>(TaskCompletionSource<T> tcs) where T : MmMessage
364 {
365 lock (_queryLock)
366 {
367 int id = _asyncQueryId++;
368 _asyncQueries[id] = tcs;
369 return id;
370 }
371 }
372
373 private static void UnregisterAsyncQuery(int queryId)
374 {
375 lock (_queryLock)
376 {
377 _asyncQueries.Remove(queryId);
378 }
379 }
380
381 private static void ResolveAsyncQuery<T>(int queryId, T response) where T : MmMessage
382 {
383 lock (_queryLock)
384 {
385 if (_asyncQueries.TryGetValue(queryId, out var obj) && obj is TaskCompletionSource<T> tcs)
386 {
387 _asyncQueries.Remove(queryId);
388 tcs.TrySetResult(response);
389 }
390 }
