205 public void SendToOtherClients(int excludeClientId, byte[] data, MmReliability reliability = MmReliability.Reliable)
206 {

208
209 // In server mode or echo mode, echo back (excluding the specified client)

218 #region Message Delivery
219
220 private void DeliverMessage(byte[] data, int senderId)
221 {
222 if (UseMessageQueue)
223 {
224 var deliveryTime = DateTime.Now.AddMilliseconds(SimulatedLatencyMs);
225 _messageQueue.Enqueue((data, senderId, deliveryTime));
226 }
227 else
228 {
229 // Immediate delivery
230 OnMessageReceived?.Invoke(data, senderId);
231 }
232 }
233
238 public void ProcessPendingMessages()
239 {
240 var now = DateTime.Now;
241 while (_messageQueue.Count > 0)
242 {
243 var (data, senderId, deliveryTime) = _messageQueue.Peek();
244 if (deliveryTime <= now)
245 {
246 _messageQueue.Dequeue();
247 OnMessageReceived?.Invoke(data, senderId);
248 }
249 else
250 {
251 break; // Next message not ready yet
252 }
253 }
254 }
255
259 public int PendingMessageCount => _messageQueue.Count;
260
264 public void FlushMessages()
265 {
266 while (_messageQueue.Count > 0)
267 {
268 var (data, senderId, _) = _messageQueue.Dequeue();
269 OnMessageReceived?.Invoke(data, senderId);
270 }
271 }
272
273 #endregion
