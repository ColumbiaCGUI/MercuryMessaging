148 public MmStateDelta ComputeDelta()
149 {
150 var changedFields = new List<MmDeltaField>();
151
152 foreach (var kvp in _currentState)
153 {
154 ushort fieldId = kvp.Key;
155 object currentValue = kvp.Value;
156
157 if (!_lastSentState.TryGetValue(fieldId, out object lastValue) ||
158 !ValuesEqual(currentValue, lastValue))
159 {
160 if (_fieldTypes.TryGetValue(fieldId, out var fieldType))
161 {
162 changedFields.Add(new MmDeltaField(
163 fieldId,
164 fieldType,
165 SerializeValue(currentValue, fieldType)
166 ));
167 }
168 }
169 }
170
171 if (changedFields.Count == 0)
172 {
173 return null;
174 }
175
176 CurrentSequence++;
177
178 return new MmStateDelta
179 {
180 DeltaId = CurrentSequence,
181 BaselineSequence = AckedSequence,
182 ChangedFields = changedFields,
183 Timestamp = Time.timeAsDouble,
184 IsFullSnapshot = false,
185 NetId = NetId
186 };
187 }
188
192 public MmStateDelta CreateFullSnapshot()
193 {
194 var allFields = new List<MmDeltaField>();
195
196 foreach (var kvp in _currentState)
197 {
198 ushort fieldId = kvp.Key;
199 if (_fieldTypes.TryGetValue(fieldId, out var fieldType))
200 {
201 allFields.Add(new MmDeltaField(
202 fieldId,
203 fieldType,
204 SerializeValue(kvp.Value, fieldType)
205 ));
206 }
207 }
208
209 CurrentSequence++;
210
211 return new MmStateDelta
212 {
213 DeltaId = CurrentSequence,
214 BaselineSequence = 0,
215 ChangedFields = allFields,
216 Timestamp = Time.timeAsDouble,
217 IsFullSnapshot = true,
218 NetId = NetId
219 };
220 }

259 private bool ValuesEqual(object a, object b)
260 {
261 if (a == null && b == null) return true;
262 if (a == null || b == null) return false;
263
264 // Special handling for Vector3 (floating point comparison)
265 if (a is Vector3 va && b is Vector3 vb)
266 {
267 return Vector3.Distance(va, vb) < 0.001f;
268 }
269
270 // Special handling for Quaternion
271 if (a is Quaternion qa && b is Quaternion qb)
272 {
273 return Mathf.Abs(Quaternion.Dot(qa, qb)) > 0.9999f;
274 }
275
276 return a.Equals(b);
277 }

279 private byte[] SerializeValue(object value, MmDeltaFieldType type)
280 {
281 using (var ms = new System.IO.MemoryStream())
282 using (var writer = new System.IO.BinaryWriter(ms))
283 {
284 switch (type)
285 {
286 case MmDeltaFieldType.Bool:
287 writer.Write((bool)value);
288 break;
289 case MmDeltaFieldType.Int:
290 writer.Write((int)value);
291 break;
292 case MmDeltaFieldType.Float:
293 writer.Write((float)value);
294 break;
295 case MmDeltaFieldType.String:
296 writer.Write((string)value ?? "");
297 break;
298 case MmDeltaFieldType.Vector3:
299 var v3 = (Vector3)value;
300 writer.Write(v3.x);
301 writer.Write(v3.y);
302 writer.Write(v3.z);
303 break;
304 case MmDeltaFieldType.Vector4:
305 var v4 = (Vector4)value;
306 writer.Write(v4.x);
307 writer.Write(v4.y);
308 writer.Write(v4.z);
309 writer.Write(v4.w);
310 break;
311 case MmDeltaFieldType.Quaternion:
312 var q = (Quaternion)value;
313 writer.Write(q.x);
314 writer.Write(q.y);
315 writer.Write(q.z);
316 writer.Write(q.w);
317 break;
318 case MmDeltaFieldType.ByteArray:
319 var bytes = (byte[])value;
320 writer.Write(bytes.Length);
321 writer.Write(bytes);
322 break;
323 case MmDeltaFieldType.Transform:
324 var t = (MmTransform)value;
325 writer.Write(t.Translation.x);
326 writer.Write(t.Translation.y);
327 writer.Write(t.Translation.z);
328 writer.Write(t.Scale.x);
329 writer.Write(t.Scale.y);
330 writer.Write(t.Scale.z);
331 writer.Write(t.Rotation.x);
332 writer.Write(t.Rotation.y);
333 writer.Write(t.Rotation.z);
334 writer.Write(t.Rotation.w);
335 break;
336 default:
337 throw new ArgumentException($"Unknown field type: {type}");
338 }
339 return ms.ToArray();
340 }
341 }

343 private object DeserializeValue(byte[] data, MmDeltaFieldType type)
344 {
345 using (var ms = new System.IO.MemoryStream(data))
346 using (var reader = new System.IO.BinaryReader(ms))
347 {
348 switch (type)
349 {
350 case MmDeltaFieldType.Bool:
351 return reader.ReadBoolean();
352 case MmDeltaFieldType.Int:
353 return reader.ReadInt32();
354 case MmDeltaFieldType.Float:
355 return reader.ReadSingle();
356 case MmDeltaFieldType.String:
357 return reader.ReadString();
358 case MmDeltaFieldType.Vector3:
359 return new Vector3(reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle());
360 case MmDeltaFieldType.Vector4:
361 return new Vector4(reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle());
362 case MmDeltaFieldType.Quaternion:
363 return new Quaternion(reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle());
364 case MmDeltaFieldType.ByteArray:
365 int len = reader.ReadInt32();
366 return reader.ReadBytes(len);
367 case MmDeltaFieldType.Transform:
368 return new MmTransform(
369 new Vector3(reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle()),
370 new Vector3(reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle()),
371 new Quaternion(reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle(), reader.ReadSingle())
372 );
373 default:
374 throw new ArgumentException($"Unknown field type: {type}");
375 }
376 }
377 }
