59 public override void Apply()
60 {
61 var f =

63 //var f = (float)_step / _steps;
64
65 ValueCurrent =
66 MatrixUtilities.CreateRotationMatrix(Quaternion.Slerp(startRotation, endRotation, f))
67 * MatrixUtilities.CreateTranslationMatrix(Vector3.Lerp(startTranstion, endTranslation, f));
68
69 transform.SetPosition(ValueCurrent.GetPosition(), SetGlobal);
70 transform.SetRotation(ValueCurrent.GetRotation(), SetGlobal);
71 }
