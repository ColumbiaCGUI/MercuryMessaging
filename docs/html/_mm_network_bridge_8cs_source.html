313 private bool ValidateSend(MmMessage message)
314 {
315 if (!IsInitialized)
316 {
317 MmLog.LogError("MmNetworkBridge: Cannot send - not initialized");
318 return false;

320 if (!Backend.IsConnected)
321 {
322 MmLog.LogError("MmNetworkBridge: Cannot send - not connected");
323 return false;

327 MmLog.LogError("MmNetworkBridge: Cannot send null message");
328 return false;
329 }

333 private void PrepareMessageForSend(MmMessage message)
334 {
335 // If this is a GameObject message, resolve the network ID
336 if (message is MmMessageGameObject goMsg && goMsg.Value != null)
337 {
338 if (Resolver.TryGetNetworkId(goMsg.Value, out uint netId))
339 {
340 goMsg.GameObjectNetId = netId;
341 }
342 else
343 {
344 MmLog.LogWarning($"MmNetworkBridge: Could not resolve network ID for {goMsg.Value.name}");
345 }
346 }
347 }

351 #region Receiving

353 private void HandleReceivedMessage(byte[] data, int senderId)
354 {
355 try
356 {
357 MmMessage message = MmBinarySerializer.Deserialize(data);
358
359 // Resolve GameObject references
360 ResolveGameObjectReferences(message);
361
362 OnMessageReceived?.Invoke(message);
363
364 // Route to the appropriate MmRelayNode
365 RouteMessage(message);
366 }
367 catch (Exception e)
368 {
369 // FAIL-FAST: Log full exception with stack trace, not just message
370 MmLog.LogError($"MmNetworkBridge: Failed to process received message from sender {senderId}");
371 Debug.LogException(e);
372 // Note: We don't rethrow here as network message processing should be resilient,
373 // but the full error is logged for debugging
374 }
375 }
376
377 private void ResolveGameObjectReferences(MmMessage message)
378 {
379 // Resolve GameObject message values
380 if (message is MmMessageGameObject goMsg && goMsg.GameObjectNetId != 0)
381 {
382 goMsg.Value = Resolver.GetGameObject(goMsg.GameObjectNetId);
383 }
384 }
385
386 private void RouteMessage(MmMessage message)
387 {
388 // Find the target relay node by NetId
389 if (message.NetId == 0)
390 {
391 MmLog.LogWarning("MmNetworkBridge: Received message with no target NetId");
392 return;
393 }
394
395 if (TryGetRelayNode(message.NetId, out MmRelayNode relayNode))
396 {
397 // Mark as deserialized so the relay node knows this came from network
398 relayNode.MmInvoke(message);
399 MmLog.LogNetwork($"MmNetworkBridge: Routed {message.MmMessageType} to {relayNode.gameObject.name}");
400 }
401 else if (Resolver.TryGetRelayNode(message.NetId, out relayNode))
402 {
403 // Fallback: try to find via resolver
404 RegisterRelayNode(message.NetId, relayNode);
405 relayNode.MmInvoke(message);
406 MmLog.LogNetwork($"MmNetworkBridge: Routed {message.MmMessageType} to {relayNode.gameObject.name} (via resolver)");
407 }
408 else
409 {
410 MmLog.LogWarning($"MmNetworkBridge: No relay node found for network ID {message.NetId}");
411 }
412 }
