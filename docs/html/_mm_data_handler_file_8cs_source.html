147 public virtual int Open(bool overwrite = false)
148 {
149 int newItem = 0;
150
151#if UNITY_STANDALONE || UNITY_EDITOR
152 FileInfo t = new FileInfo(Path.Combine(FileLocation, FileName));
153
154 newItem = (!t.Exists || overwrite) ? 1 : 0;
155
156 if (overwrite && t.Exists)
157 {
158 t.Delete();
159 MmLogger.LogApplication("File Deleted: " + FileName);
160 }
161 Writer = t.AppendText();
162 MmLogger.LogApplication("File opened: " + FileName);
163#endif
164
165 return newItem;
166 }
