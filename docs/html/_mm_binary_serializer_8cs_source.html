89 public static byte[] Serialize(MmMessage message)
90 {
91 using (var stream = new MemoryStream())
92 using (var writer = new BinaryWriter(stream))
93 {
94 // Write header
95 writer.Write(MagicNumber);
96 writer.Write(FormatVersion);
97 writer.Write((short)message.MmMessageType);
98 writer.Write((short)message.MmMethod);
99 writer.Write(message.NetId);
100 writer.Write(PackMetadata(message.MetadataBlock)); // uint (4 bytes)
101
102 // Write type-specific payload
103 WritePayload(writer, message);
104
105 return stream.ToArray();
106 }
107 }

115 public static byte[] SerializePooled(MmMessage message)
116 {

121 writer.WriteByte(FormatVersion);
122 writer.WriteShort((short)message.MmMessageType);
123 writer.WriteShort((short)message.MmMethod);
124 writer.WriteUInt(message.NetId);
125 writer.WriteUInt(PackMetadata(message.MetadataBlock)); // uint (4 bytes)
126
127 // Write type-specific payload
128 WritePayloadPooled(writer, message);
129
130 return writer.ToArray();

221 private static void WriteSerializableDataPooled(MmWriter writer, MmMessageSerializable msg)
222 {
223 // Check for new IMmBinarySerializable first (zero-allocation path)
224 if (msg.value is IMmBinarySerializable binarySerializable)
225 {
226 // Write type ID from registry
227 ushort typeId = MmTypeRegistry.GetTypeId(binarySerializable);
228 writer.WriteUShort(typeId);
229 if (typeId != MmTypeRegistry.NullTypeId)
230 {
231 binarySerializable.WriteTo(writer);
232 }
233 return;
234 }
235
236 // Legacy path: IMmSerializable with object[] (fallback)
237 string typeName = msg.value?.GetType().AssemblyQualifiedName ?? "";
238 writer.WriteString(typeName);
239
240 if (msg.value is MercuryMessaging.Task.IMmSerializable serializable)
241 {
242 object[] data = serializable.Serialize();
243 writer.WriteInt(data.Length);
244 foreach (var obj in data)
245 {
246 WriteObjectValuePooled(writer, obj);
247 }
248 }
249 else if (msg.value != null)
250 {
251 Debug.LogWarning($"MmBinarySerializer: Type {typeName} does not implement IMmBinarySerializable or IMmSerializable.");
252 writer.WriteInt(0);
253 }
254 else
255 {
256 writer.WriteInt(0);
257 }

260 private static void WriteObjectValuePooled(MmWriter writer, object value)
261 {
262 if (value == null)
263 {
264 writer.WriteByte(0);
265 return;
266 }
267
268 switch (value)
269 {
270 case int i:
271 writer.WriteByte(1);
272 writer.WriteInt(i);
273 break;
274 case bool b:
275 writer.WriteByte(2);
276 writer.WriteBool(b);
277 break;
278 case float f:
279 writer.WriteByte(3);
280 writer.WriteFloat(f);
281 break;
282 case string s:
283 writer.WriteByte(4);
284 writer.WriteString(s);
285 break;
286 case double d:
287 writer.WriteByte(5);
288 writer.WriteDouble(d);
289 break;
290 case long l:
291 writer.WriteByte(6);
292 writer.WriteLong(l);
293 break;
294 case byte by:
295 writer.WriteByte(7);
296 writer.WriteByte(by);
297 break;
298 case short sh:
299 writer.WriteByte(8);
300 writer.WriteShort(sh);
301 break;

304 writer.WriteByte(0);

317 private static uint PackMetadata(MmMetadataBlock metadata)
318 {
319 uint packed = 0;
320 packed |= ((uint)metadata.LevelFilter & 0x0F);
321 packed |= ((uint)metadata.ActiveFilter & 0x03) << 4;
322 packed |= ((uint)metadata.SelectedFilter & 0x03) << 6;
323 packed |= ((uint)metadata.NetworkFilter & 0x03) << 8;
324 // Tag uses 8 bits (supports Tag0-Tag7, where Tag7 = 0x80)
325 packed |= ((uint)metadata.Tag & 0xFF) << 10;
326 return packed;
327 }

404 public static MmMessage Deserialize(byte[] data)
405 {
406 if (data == null || data.Length < HeaderSize)
407 {
408 throw new ArgumentException("Invalid message data: too short");
409 }
410
411 using (var stream = new MemoryStream(data))
412 using (var reader = new BinaryReader(stream))
413 {
414 // Validate magic number
415 byte[] magic = reader.ReadBytes(4);
416 if (!ValidateMagic(magic))
417 {
418 throw new InvalidDataException("Invalid message: bad magic number");
419 }
420
421 // Read version
422 byte version = reader.ReadByte();
423 if (version > FormatVersion)
424 {
425 throw new InvalidDataException($"Unsupported message version: {version}");
426 }
427
428 // Read header
429 var messageType = (MmMessageType)reader.ReadInt16();
430 var method = (MmMethod)reader.ReadInt16();
431 uint netId = reader.ReadUInt32();
432 uint packedMetadata = reader.ReadUInt32();
433 var metadata = UnpackMetadata(packedMetadata);
434
435 // Create and populate message
436 MmMessage message = CreateMessage(messageType, metadata);
437 message.MmMethod = method;
438 message.NetId = netId;
439
440 // Read type-specific payload
441 ReadPayload(reader, message);
442
443 return message;
444 }
445 }

453 public static MmMessage DeserializePooled(byte[] data)
454 {
455 if (data == null || data.Length < HeaderSize)
456 {
457 throw new ArgumentException("Invalid message data: too short");
458 }
459
460 var reader = new MmReader(data);
461
462 // Validate magic number
463 byte m0 = reader.ReadByte();
464 byte m1 = reader.ReadByte();
465 byte m2 = reader.ReadByte();
466 byte m3 = reader.ReadByte();
467 if (m0 != MagicNumber[0] || m1 != MagicNumber[1] ||
468 m2 != MagicNumber[2] || m3 != MagicNumber[3])
469 {
470 throw new InvalidDataException("Invalid message: bad magic number");
471 }
472
473 // Read version
474 byte version = reader.ReadByte();
475 if (version > FormatVersion)
476 {
477 throw new InvalidDataException($"Unsupported message version: {version}");
478 }
479
480 // Read header
481 var messageType = (MmMessageType)reader.ReadShort();
482 var method = (MmMethod)reader.ReadShort();
483 uint netId = reader.ReadUInt();
484 uint packedMetadata = reader.ReadUInt();
485 var metadata = UnpackMetadata(packedMetadata);
486
487 // Create and populate message
488 MmMessage message = CreateMessage(messageType, metadata);
489 message.MmMethod = method;
490 message.NetId = netId;
491
492 // Read type-specific payload
493 ReadPayloadPooled(reader, message);
494
495 return message;
496 }

581 private static void ReadSerializableDataPooled(MmReader reader, MmMessageSerializable msg)
582 {
583 // Try new IMmBinarySerializable path first (check for type ID)
584 ushort typeId = reader.ReadUShort();
585 if (typeId != MmTypeRegistry.NullTypeId)
586 {
587 // New binary serialization path
588 var instance = MmTypeRegistry.Create(typeId);
589 if (instance != null)
590 {
591 instance.ReadFrom(reader);
592 msg.value = instance;
593 }
594 return;
595 }
596
597 // Legacy path: type name + object array
598 string typeName = reader.ReadString();
599 int dataLength = reader.ReadInt();
600
601 if (string.IsNullOrEmpty(typeName))
602 {
603 return;
604 }
605
606 // Read the object array
607 object[] data = new object[dataLength];
608 for (int i = 0; i < dataLength; i++)
609 {
610 data[i] = ReadObjectValuePooled(reader);
611 }
612
613 try
614 {
615 Type type = Type.GetType(typeName);
616 if (type == null)
617 {
618 Debug.LogError($"MmBinarySerializer: Cannot find type '{typeName}' for deserialization");
619 return;
620 }
621
622 var obj = Activator.CreateInstance(type);
623 if (obj is MercuryMessaging.Task.IMmSerializable serializable)
624 {
625 serializable.Deserialize(data, 0);
626 msg.value = serializable;
627 }
628 else
629 {
630 Debug.LogError($"MmBinarySerializer: Type '{typeName}' does not implement IMmSerializable");
631 }
632 }
633 catch (Exception e)
634 {
635 Debug.LogException(e);
636 throw new InvalidOperationException($"Failed to deserialize MmMessageSerializable of type '{typeName}'", e);
637 }
638 }

640 private static object ReadObjectValuePooled(MmReader reader)
641 {
642 byte typeMarker = reader.ReadByte();
643 switch (typeMarker)
644 {
645 case 0: return null;
646 case 1: return reader.ReadInt();
647 case 2: return reader.ReadBool();
648 case 3: return reader.ReadFloat();
649 case 4: return reader.ReadString();
650 case 5: return reader.ReadDouble();
651 case 6: return reader.ReadLong();
652 case 7: return reader.ReadByte();
653 case 8: return reader.ReadShort();
654 default:
655 Debug.LogWarning($"MmBinarySerializer: Unknown type marker {typeMarker} during deserialization");
656 return null;
657 }
658 }

663 private static bool ValidateMagic(byte[] magic)
664 {
665 if (magic.Length != 4) return false;
666 for (int i = 0; i < 4; i++)
667 {
668 if (magic[i] != MagicNumber[i]) return false;
669 }
670 return true;
671 }

676 private static MmMetadataBlock UnpackMetadata(uint packed)
677 {
678 var level = (MmLevelFilter)(packed & 0x0F);
679 var active = (MmActiveFilter)((packed >> 4) & 0x03);
680 var selected = (MmSelectedFilter)((packed >> 6) & 0x03);
681 var network = (MmNetworkFilter)((packed >> 8) & 0x03);
682 var tag = (MmTag)((packed >> 10) & 0xFF); // 8 bits for all Tag0-Tag7
683
684 return new MmMetadataBlock(tag, level, active, selected, network);
685 }

865 private static void WriteSerializableData(BinaryWriter writer, MmMessageSerializable msg)
866 {
867 // Write the type name for deserialization
868 string typeName = msg.value?.GetType().AssemblyQualifiedName ?? "";
869 WriteString(writer, typeName);
870
871 if (msg.value is MercuryMessaging.Task.IMmSerializable serializable)
872 {
873 // Use the proper IMmSerializable interface
874 object[] data = serializable.Serialize();
875 writer.Write(data.Length);
876 foreach (var obj in data)
877 {
878 WriteObjectValue(writer, obj);
879 }
880 }
881 else if (msg.value != null)
882 {
883 // Fallback for non-IMmSerializable types - log warning
884 Debug.LogWarning($"MmBinarySerializer: Type {typeName} does not implement IMmSerializable. Serialization may fail.");
885 writer.Write(0); // Empty array
886 }
887 else
888 {
889 writer.Write(0); // Null value - empty array
890 }

896 private static void WriteObjectValue(BinaryWriter writer, object value)
897 {
898 if (value == null)
899 {
900 writer.Write((byte)0); // Null type marker
901 return;
902 }
903
904 switch (value)
905 {
906 case int i:
907 writer.Write((byte)1);
908 writer.Write(i);
909 break;
910 case bool b:
911 writer.Write((byte)2);
912 writer.Write(b);
913 break;
914 case float f:
915 writer.Write((byte)3);
916 writer.Write(f);
917 break;
918 case string s:
919 writer.Write((byte)4);
920 WriteString(writer, s);
921 break;
922 case double d:
923 writer.Write((byte)5);

1022 private static void ReadSerializableData(BinaryReader reader, MmMessageSerializable msg)
1023 {
1024 string typeName = ReadString(reader);
1025 int dataLength = reader.ReadInt32();
1026
1027 if (string.IsNullOrEmpty(typeName))
1028 {
1029 return; // No type, nothing to deserialize
1030 }
1031
1032 // Read the object array
1033 object[] data = new object[dataLength];
1034 for (int i = 0; i < dataLength; i++)
1035 {
1036 data[i] = ReadObjectValue(reader);
1037 }
1038
1039 try
1040 {
1041 Type type = Type.GetType(typeName);
1042 if (type == null)
1043 {
1044 Debug.LogError($"MmBinarySerializer: Cannot find type '{typeName}' for deserialization");
1045 return;
1046 }
1047
1048 var obj = Activator.CreateInstance(type);
1049 if (obj is MercuryMessaging.Task.IMmSerializable serializable)
1050 {
1051 // Call Deserialize with the object array
1052 serializable.Deserialize(data, 0);
1053 msg.value = serializable;
1054 }
1055 else
1056 {
1057 Debug.LogError($"MmBinarySerializer: Type '{typeName}' does not implement IMmSerializable");
1058 }
1059 }
1060 catch (Exception e)
1061 {
1062 // FAIL-FAST: Log full exception, not just message
1063 Debug.LogException(e);
1064 throw new InvalidOperationException($"Failed to deserialize MmMessageSerializable of type '{typeName}'", e);
1065 }
1066 }

1071 private static object ReadObjectValue(BinaryReader reader)
1072 {
1073 byte typeMarker = reader.ReadByte();
1074 switch (typeMarker)
1075 {
1076 case 0: return null;
1077 case 1: return reader.ReadInt32();
1078 case 2: return reader.ReadBoolean();
1079 case 3: return reader.ReadSingle();
1080 case 4: return ReadString(reader);
1081 case 5: return reader.ReadDouble();
1082 case 6: return reader.ReadInt64();
1083 case 7: return reader.ReadByte();
1084 case 8: return reader.ReadInt16();
1085 default:
1086 Debug.LogWarning($"MmBinarySerializer: Unknown type marker {typeMarker} during deserialization");
1087 return null;
1088 }
1089 }
