66 public static void Register<T>(ushort typeId) where T : IMmBinarySerializable, new()
67 {
68 if (typeId == NullTypeId)
69 {
70 throw new ArgumentException("Type ID 0 is reserved for null types", nameof(typeId));
71 }
72
73 Type type = typeof(T);
74
75 // Check for duplicate ID
76 if (_factories.ContainsKey(typeId))
77 {
78 if (_idToType[typeId] == type)
79 {
80 // Same type, same ID - ignore duplicate registration
81 return;
82 }
83 throw new InvalidOperationException(
84 $"Type ID {typeId} is already registered to {_idToType[typeId].Name}. " +
85 $"Cannot register {type.Name} with the same ID.");
86 }
87
88 // Check for duplicate type
89 if (_typeToId.ContainsKey(type))
90 {
91 throw new InvalidOperationException(
92 $"Type {type.Name} is already registered with ID {_typeToId[type]}. " +
93 $"Cannot register with ID {typeId}.");
94 }
95
96 _factories[typeId] = () => new T();
97 _typeToId[type] = typeId;
98 _idToType[typeId] = type;
99 }

105 public static void Register<T>(ushort typeId, Func<T> factory) where T : IMmBinarySerializable
106 {
107 if (typeId == NullTypeId)
108 {
109 throw new ArgumentException("Type ID 0 is reserved for null types", nameof(typeId));
110 }
111
112 if (factory == null)
113 {
114 throw new ArgumentNullException(nameof(factory));
115 }
116
117 Type type = typeof(T);
118
119 if (_factories.ContainsKey(typeId))
120 {
121 if (_idToType[typeId] == type) return; // Ignore duplicate
122 throw new InvalidOperationException(
123 $"Type ID {typeId} is already registered to {_idToType[typeId].Name}");
124 }
125
126 if (_typeToId.ContainsKey(type))
127 {
128 throw new InvalidOperationException(
129 $"Type {type.Name} is already registered with ID {_typeToId[type]}");
130 }
131
132 _factories[typeId] = () => factory();
133 _typeToId[type] = typeId;
134 _idToType[typeId] = type;
135 }

247 public static IMmBinarySerializable ReadFrom(MmReader reader)
248 {
249 ushort typeId = reader.ReadUShort();
250 if (typeId == NullTypeId) return null;
251
252 IMmBinarySerializable instance = Create(typeId);
253 if (instance == null)
254 {
255 // Type not registered - skip reading (data will be corrupted)
256 Debug.LogError($"MmTypeRegistry: Cannot deserialize unknown type ID {typeId}");
257 return null;
258 }
259
260 instance.ReadFrom(reader);
261 return instance;

268 public static void WriteTo(MmWriter writer, IMmBinarySerializable instance)
269 {
270 if (instance == null)
271 {
272 writer.WriteUShort(NullTypeId);
273 return;
274 }
275
276 ushort typeId = GetTypeId(instance);
277 if (typeId == NullTypeId)
278 {
279 Debug.LogWarning($"MmTypeRegistry: Type {instance.GetType().Name} is not registered. " +
280 "Serialization will write null type ID.");
281 }
282
283 writer.WriteUShort(typeId);
284 if (typeId != NullTypeId)
285 {
286 instance.WriteTo(writer);
287 }
