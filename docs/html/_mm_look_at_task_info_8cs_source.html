105 public override int Parse(string str)
106 {
107 base.Parse(str);
108
109 var words = str.Split(',');
110
111 Position = TransformExtensions.CSV2Vector3(words[16], '|');
112 LookAt = TransformExtensions.CSV2Vector3(words[17], '|');
113 TargetObjOrientation = TransformExtensions.CSV2Quaternion(words[18], '|');
114
115 CalculateOrientation();
116
117 return 19;

132 public override int Deserialize(object[] data, int index)
133 {

137 TargetObjOrientation = new Quaternion(
138 (float) data[index++],
139 (float) data[index++],
140 (float) data[index++],
141 (float) data[index++]
142 );
143 CalculateOrientation();
144 return index;
145 }
146
151 public override object[] Serialize()
