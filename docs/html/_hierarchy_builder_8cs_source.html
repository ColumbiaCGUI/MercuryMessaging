201 private MmRelayNode BuildNode(NodeBuilder builder, Transform parent)
202 {
203 GameObject gameObject;
204 MmRelayNode relay;
205
206 // If using existing relay node
207 if (builder.ExistingRelay != null)
208 {
209 relay = builder.ExistingRelay;
210 gameObject = relay.gameObject;
211 }
212 else
213 {
214 // Create new GameObject
215 gameObject = new GameObject(builder.Name);
216
217 if (parent != null)
218 {
219 gameObject.transform.SetParent(parent);
220 }
221
222 // Add relay node (switch or regular)
223 if (builder.IsSwitchNode)
224 {
225 relay = gameObject.AddComponent<MmRelaySwitchNode>();
226 }
227 else
228 {
229 relay = gameObject.AddComponent<MmRelayNode>();
230 }
231 }
232
233 // Apply configuration
234 if (builder.Tag.HasValue)
235 {
236 relay.Tag = builder.Tag.Value;
237 }
238
239 if (builder.Layer.HasValue)
240 {
241 relay.layer = builder.Layer.Value;
242 }
243
244 if (builder.Position.HasValue)
245 {
246 gameObject.transform.localPosition = builder.Position.Value;
247 }
248
249 // Add responder component
250 if (builder.ResponderType != null)
251 {
252 gameObject.AddComponent(builder.ResponderType);
253 }
254
255 // Run configure action
256 builder.ConfigureAction?.Invoke(relay);
257
258 // Set active state
259 if (!builder.StartActive)
260 {
261 gameObject.SetActive(false);
262 }
263
264 // Build children
265 foreach (var childBuilder in builder.Children)
266 {
267 var childRelay = BuildNode(childBuilder, gameObject.transform);
268
269 // Register child in routing table
270 relay.MmAddToRoutingTable(childRelay, MmLevelFilter.Child);
271 childRelay.AddParent(relay);
272 }
273
274 // Refresh responders
275 relay.MmRefreshResponders();
276
277 return relay;
278 }

452 public static void MmSetParent(this MmRelayNode node, MmRelayNode newParent, bool worldPositionStays = true)
453 {
454 if (node == null) return;
455
456 // Step 1: Unregister from old parent's routing table
457 var oldParentTransform = node.transform.parent;
458 if (oldParentTransform != null)
459 {
460 var oldParentRelay = oldParentTransform.GetComponentInParent<MmRelayNode>();
461 if (oldParentRelay != null)
462 {
463 // Find and remove this node from old parent's routing table
464 for (int i = oldParentRelay.RoutingTable.Count - 1; i >= 0; i--)
465 {
466 if (oldParentRelay.RoutingTable[i].Responder == node)
467 {
468 oldParentRelay.RoutingTable.RemoveAt(i);
469 break;
470 }
471 }
472 }
473 }
474
475 // Step 2: Unity hierarchy update
476 node.transform.SetParent(newParent?.transform, worldPositionStays);
477
478 // Step 3: Register with new parent (if any)
479 if (newParent != null)
480 {
481 newParent.MmAddToRoutingTable(node, MmLevelFilter.Child);
482 node.AddParent(newParent);
483 }
484
485 // Step 4: Refresh parent list to update internal state
486 node.RefreshParents();
487 }
488
495 public static void MmSetParent(this MmRelayNode node, Transform newParentTransform, bool worldPositionStays = true)
496 {
497 if (node == null) return;
498
499 MmRelayNode newParentRelay = null;
500 if (newParentTransform != null)
501 {
502 newParentRelay = newParentTransform.GetComponent<MmRelayNode>();
503 if (newParentRelay == null)
504 {
505 Debug.LogWarning($"[MmSetParent] Target transform '{newParentTransform.name}' has no MmRelayNode. " +
506 "Using Unity SetParent only - message routing will not work.", node);
507 node.transform.SetParent(newParentTransform, worldPositionStays);
508 return;
509 }
510 }
511
512 node.MmSetParent(newParentRelay, worldPositionStays);
513 }
