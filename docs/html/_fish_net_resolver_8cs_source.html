62 public bool TryGetNetworkId(GameObject gameObject, out uint networkId)
63 {
64#if FISHNET_AVAILABLE
65 if (gameObject != null)
66 {
67 var nob = gameObject.GetComponent<NetworkObject>();
68 // Reject ObjectId 0 - FishNet uses 0 for scene objects, but MercuryMessaging
69 // treats 0 as "no target". Force fallback to deterministic ID for scene objects.
70 if (nob != null && nob.IsSpawned && nob.ObjectId > 0)
71 {
72 networkId = (uint)nob.ObjectId;
73 return true;
74 }
75 }
76#endif
77 networkId = 0;
78 return false;
79 }
80
82 public bool TryGetGameObject(uint networkId, out GameObject gameObject)
83 {
84#if FISHNET_AVAILABLE
85 // Try server-side spawned objects first
86 if (InstanceFinder.IsServerStarted &&
87 InstanceFinder.ServerManager != null &&
88 InstanceFinder.ServerManager.Objects != null)
89 {
90 if (InstanceFinder.ServerManager.Objects.Spawned
91 .TryGetValue((int)networkId, out NetworkObject serverNob))
92 {
93 gameObject = serverNob.gameObject;
94 return true;
95 }
96 }
97
98 // Try client-side spawned objects
99 if (InstanceFinder.IsClientStarted &&
100 InstanceFinder.ClientManager != null &&
101 InstanceFinder.ClientManager.Objects != null)
102 {
103 if (InstanceFinder.ClientManager.Objects.Spawned
104 .TryGetValue((int)networkId, out NetworkObject clientNob))
105 {
106 gameObject = clientNob.gameObject;
107 return true;
108 }
109 }
110#endif
111 gameObject = null;
112 return false;
113 }
