110 public virtual int LoadPartiallyFinishedTaskSequence()
111 {
112 if (!File.Exists(Path.Combine(MmTaskUserData.DirPath, PartialDataFile)))
113 return 0;
114
115 var reader = new StreamReader(Path.Combine(MmTaskUserData.DirPath,
116 PartialDataFile),
117 Encoding.UTF8);
118
119 string line = reader.ReadLine();
120 if (line == null)
121 {
122 MmLogger.LogError("Partial Sequence File Empty");
123 return -1;
124 }
125
126 var curSequenceIndex = int.Parse(line.Split(',')[1]);
127
128 reader.Close();
129
130 return curSequenceIndex;
131 }
132
139 public virtual void LoadTaskSequence(ref LinkedList<U> taskInfos)
140 {
141 var filename = MmTaskUserData.BaseDirectory + "/" + TaskSeqFilename;
142
143 try
144 {
145 using (var sr = new StreamReader(filename, Encoding.UTF8))
146 {
147 // Skip header line
148 var line = sr.ReadLine();
149
150 if (string.IsNullOrEmpty(line)) return;
151
152 taskInfos = new LinkedList<U>();
153
154 while ((line = sr.ReadLine()) != null)
155 {
156 var t = new U();
157 t.Parse(line);
158
159 // Only load current user
160 if (t.UserId != MmTaskUserData.UserId) continue;
161
162 taskInfos.AddLast(t);
163 }
164 }
165 }
166 catch (Exception e)
167 {
168 MmLogger.LogError("The file could not be read:");
169 MmLogger.LogError(e.Message);
170 }
171 }
172

181
182 writer.WriteLine("Current Sequence ID," + seqVal);
183 //TODO: Check if UserSequence is the right variable to write
184
185 writer.Close();
186 }
