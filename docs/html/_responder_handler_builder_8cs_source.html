62 public ResponderHandlerBuilder OnCustomMethod(int methodId, Action<MmMessage> handler)
63 {
64 if (methodId < 1000)
65 throw new ArgumentException("Custom methods must be >= 1000", nameof(methodId));
66 if (handler == null)
67 throw new ArgumentNullException(nameof(handler));
68
69 _customHandlers.Add(((MmMethod)methodId, handler));
70 return this;
71 }
72
76 public ResponderHandlerBuilder OnCustomMethod<T>(int methodId, Action<T> handler) where T : MmMessage
77 {
78 if (handler == null)
79 throw new ArgumentNullException(nameof(handler));
80
81 return OnCustomMethod(methodId, msg => handler((T)msg));
82 }

225 public ResponderHandlerWrapper Build()
226 {
227 // Register custom handlers using reflection to access protected method
228 foreach (var (method, handler) in _customHandlers)
229 {
230 // Use reflection to call protected RegisterCustomHandler
231 var registerMethod = typeof(MmExtendableResponder)
232 .GetMethod("RegisterCustomHandler",
233 System.Reflection.BindingFlags.Instance | System.Reflection.BindingFlags.NonPublic);
234
235 registerMethod?.Invoke(_responder, new object[] { method, handler });
236 }
237
238 // Return wrapper with standard method callbacks
239 return new ResponderHandlerWrapper(
240 _responder,
241 _onInitialize,
242 _onRefresh,
243 _onSetActive,
244 _onSwitch,
245 _onComplete,
246 _onMessage,
247 _onMessageBool,
248 _onMessageInt,
249 _onMessageFloat,
250 _onMessageString,
251 _onMessageVector3
252 );
253 }

255 #endregion
