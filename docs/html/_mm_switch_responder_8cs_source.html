60 public override void Awake()
61 {
62 MmLogger.LogResponder("MmSwitchResponder Awake");
63
64 MmRelaySwitchNode = GetComponent <MmRelaySwitchNode>();
65
66 //This is to avoid the situation where the MmRelaySwitchNode is started
67 // before this script is.
68 if(MmRelaySwitchNode.Initialized)
69 {
70 OnMmSwitchNodeAwakeCompleteCallback ();
71 }
72 else
73 {
74 MmRelaySwitchNode.MmRegisterAwakeCompleteCallback(OnMmSwitchNodeAwakeCompleteCallback);
75 }
76
77 MmRelaySwitchNode.MmRegisterStartCompleteCallback(OnMmNodeSwitchStartCompleteCallback);
78
79 if (MmRelaySwitchNode.MmNetworkResponder == null ||
80 MmRelaySwitchNode.MmNetworkResponder.IsActiveAndEnabled)
81 {
82 MmRegisterStartCompleteCallback(MmSwitchSetup);
83 }
84 else
85 {
86 MmRelaySwitchNode.MmNetworkResponder.MmRegisterStartCompleteCallback(MmSwitchSetup);
87 }
88
89 base.Awake ();
90 }
91
95 public override void Start()
96 {
97 MmLogger.LogResponder("MmSwitchResponder Start");
98
99 // Assign default responder for transitioning between MmRelayNodes
100 MmRelaySwitchNode.RespondersFSM.GlobalExit += delegate
101 {
102 MmLogger.LogApplication("MmRelaySwitchNode GlobalExit");
103
104 MmRelaySwitchNode.Current.MmInvoke(MmMethod.SetActive, false,
105 new MmMetadataBlock(MmLevelFilterHelper.Default, MmActiveFilter.All,
106 default(MmSelectedFilter), MmNetworkFilter.Local));
107 };
108
109 MmRelaySwitchNode.RespondersFSM.GlobalEnter += delegate
110 {
111 MmLogger.LogApplication("MmRelaySwitchNode GlobalEnter");
112
113 MmRelaySwitchNode.Current.MmInvoke(MmMethod.SetActive, true,
114 new MmMetadataBlock(MmLevelFilterHelper.Default, MmActiveFilter.All,
115 default(MmSelectedFilter), MmNetworkFilter.Local));
116 };
117
118 base.Start();
119 }
120
124 public virtual void OnMmSwitchNodeAwakeCompleteCallback()
125 {
126 MmLogger.LogResponder("OnMmSwitchNodeAwakeComplete invoked");
127 }
128
129
133 public virtual void OnMmNodeSwitchStartCompleteCallback()
134 {
135 MmLogger.LogResponder("OnMmNodeSwitchStartComplete invoked");
136
137 if (MmRelaySwitchNode.RoutingTable.Count == 0) {
138 MmLogger.LogResponder("No MmRelayNodes in RespondersFSM on: " + gameObject.name);
139 }
140 }
141
148 public virtual void MmSwitchSetup()
149 {
150 MmLogger.LogResponder(gameObject.name + " MmSwitchSetup Invoked.");
151
152 MmRelaySwitchNode.MmInvoke(MmMethod.Initialize,
153 new MmMetadataBlock(MmLevelFilter.Child, MmActiveFilter.All,
154 default(MmSelectedFilter), MmNetworkFilter.Local));
155
156 MmRelaySwitchNode.MmInvoke(MmMethod.SetActive, false,
157 new MmMetadataBlock(MmLevelFilter.Child, MmActiveFilter.All,
158 default(MmSelectedFilter), MmNetworkFilter.Local));
159
160 if (InitialState != "")
161 {
162 MmLogger.LogResponder(gameObject.name + " attempting to jump to: " + InitialState);
163 MmRelaySwitchNode.MmInvoke(MmMethod.Switch,
164 MmRelaySwitchNode.RoutingTable[InitialState].Name,
165 new MmMetadataBlock(MmLevelFilter.Self, MmActiveFilter.All,
166 default(MmSelectedFilter), MmNetworkFilter.Local));
167 }
168 }
169
176 protected override void Switch(string iName)
177 {
178 if (MmRelaySwitchNode) // This can be null if gameObject has not awoken
179 MmRelaySwitchNode.JumpTo(iName);
180 //Debug.Log ("Jumping to: " + iName);
181
182 base.Switch (iName);
183 }
184 }
