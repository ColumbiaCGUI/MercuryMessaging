61 public void Insert(int index, T item)
62 {
63 if (index != 0)
64 throw new System.ArgumentException("MmCircularBuffer only supports Insert(0, item)", nameof(index));
65
66 if (_size == _capacity)
67 {
68 // When full: replace the oldest item in place (don't advance head)
69 int oldestPos = (_capacity + _head - _size) % _capacity;
70 _buffer[oldestPos] = item;
71 // _head and _size remain unchanged
72 }
73 else
74 {
75 // When not full: insert before current oldest without overwriting
76 int oldestPos = (_capacity + _head - _size) % _capacity;
77 int newOldestPos = (oldestPos - 1 + _capacity) % _capacity;
78 _buffer[newOldestPos] = item;
79 _size++;
80 // _head remains unchanged as it still points to next write position
81 }
82 }
