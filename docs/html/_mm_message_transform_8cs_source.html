138 public override object[] Serialize()
139 {
140 object[] baseSerialized = base.Serialize();
141 object[] transformSerialized = MmTransform.Serialize();
142
143 // Pre-allocate combined array: base + transform + 1 (LocalTransform)
144 object[] result = new object[baseSerialized.Length + transformSerialized.Length + 1];
145
146 // Copy base data using Array.Copy (no LINQ)
147 Array.Copy(baseSerialized, 0, result, 0, baseSerialized.Length);
148
149 // Copy transform data
150 Array.Copy(transformSerialized, 0, result, baseSerialized.Length, transformSerialized.Length);
151
152 // Fill LocalTransform directly
153 result[baseSerialized.Length + transformSerialized.Length] = LocalTransform;
154
155 return result;
